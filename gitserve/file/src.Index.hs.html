    9  import Data.Text (Text, unpack)
   10  import Path (toFilePath)

   12  import Text.Ginger.GVal (GVal, toGVal)
   13  import Text.Ginger.Html (Html, htmlSource)
   14  import Text.Ginger.Parse (SourcePos)
   15  import Text.Ginger.Run (Run, easyRenderM)
   16  
   17  import Env (Env (..))

   21  {-
   22  This creates the main index files from the index templates, using information from all
   23  configured respositories.
   24  -}
   25  runIndex :: Env -> [Repo] -> IO ()
   26  runIndex env repos = mapM_ (runIndexFile env repos) . envIndexTemplates $ env
   27  
   28  runIndexFile :: Env -> [Repo] -> Template -> IO ()
   29  runIndexFile env repos template = do
   30      let output = combine (toFilePath . envOutput $ env) . toFilePath . templatePath $ template
   31      unless (envQuiet env) . putStrLn $ "Writing " <> output
   32      writeFile output "" -- Clear contents of file if it exists
   33      void $
   34          easyRenderM
   35              (appendFile output . unpack . htmlSource)
   36              (packageIndex env repos)
   37              (templateGinger template)
   38  
   39  {-
   40  This packages the variables that are available inside the index scope.
   41  -}
   42  packageIndex ::
   43      Env ->
   44      [Repo] ->
   45      HashMap.HashMap Text (GVal (Run SourcePos IO Html))
   46  packageIndex env repos =
   47      HashMap.fromList
   48          [ ("host", toGVal $ envHost env)
   49          , ("repositories", toGVal repos)
   50          ]
