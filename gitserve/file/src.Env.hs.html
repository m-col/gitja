   46  {-
   47  The Config data type represents the configuration options available in the config file.
   48  Each record of the type, without the 'conf' prefix, is an option.

   50  data Config = Config

   52      , confScan :: Bool
   53      , confTemplate :: FilePath
   54      , confOutput :: FilePath
   55      , confHost :: Text
   56      }
   57      deriving stock (Generic)
   58      deriving
   59          (FromDhall)
   60          via Codec (Field (CamelCase <<< DropPrefix "conf")) Config

   62  getConfig :: String -> IO Config
   63  getConfig = input auto . pack <=< makeAbsolute

   65  {-
   66  The Env data type represents all of the program's state, including user configuration
   67  and loaded template data. This can be accessed as immutable global state at any point.
   68  -}
   69  data Env = Env
   70      { envConfig :: Config
   71      , envIndexTemplates :: [Template]
   72      , envCommitTemplate :: Maybe Template
   73      , envFileTemplate :: Maybe Template
   74      , envRepoTemplates :: [Template]
   75      , envOutput :: Path Abs Dir
   76      , envRepos :: [Path Abs Dir]
   77      , envHost :: T.Text
   78      , envQuiet :: Bool
   79      , envForce :: Bool
   80      , envRepoCopyStatics :: Path Abs Dir -> IO ()
   81      }

   83  {-
   84  This creates the runtime environment, collecting the config and loading template data
   85  from the template directory.
   86  -}
   87  loadEnv :: Bool -> Bool -> Config -> IO Env
   88  loadEnv quiet force config = do
   89      -- First ensure that the output directory exists
   90      output <- parseAbsDir <=< canonicalizePath . confOutput $ config
   91      ensureDir output
   92  
   93      -- Parse repos for env
   94      repos <-
   95          if confScan config
   96              then do
   97                  ps <- fmap concat . mapM (fmap fst . ls) . confRepos $ config
   98                  return . filter ((/=) ".git" . toFilePath . dirname) $ ps
   99              else mapM (parseAbsDir <=< canonicalizePath) . confRepos $ config
  100  
  101      -- Find template files, copying the static files as is
  102      (dirs, files) <- ls . confTemplate $ config
  103      (dirsRepo, filesRepo) <- ls $ confTemplate config FP.</> "repo"
  104      copyStaticDirs dirs output
  105      copyStaticFiles files output
  106  
  107      -- Load files from template directory
  108      indexT <- collectTemplates files
  109      commitT <- findTemplate "commit.html" filesRepo
  110      fileT <- findTemplate "file.html" filesRepo
  111      repoT <-
  112          collectTemplates
  113              . filter (flip notElem ["commit.html", "file.html"] . toFilePath . filename)
  114              $ filesRepo
  115  
  116      -- Exit early if we didn't find any templates
  117      when
  118          ( null indexT
  119              && isNothing commitT
  120              && isNothing fileT
  121              && null repoT
  122          )
  123          $ die "No templates were found."
  124  
  125      -- App environment
  126      return
  127          Env
  128              { envConfig = config
  129              , envIndexTemplates = indexT
  130              , envCommitTemplate = commitT
  131              , envFileTemplate = fileT
  132              , envRepoTemplates = repoT
  133              , envOutput = output
  134              , envRepos = repos
  135              , envHost = confHost config
  136              , envQuiet = quiet
  137              , envForce = force
  138              , envRepoCopyStatics = \p -> copyStaticDirs dirsRepo p >> copyStaticFiles filesRepo p
  139              }
  140    where
  141      ls :: FilePath -> IO ([Path Abs Dir], [Path Abs File])
  142      ls dir = do
  143          canon <- parseAbsDir =<< canonicalizePath dir
  144          exists <- doesDirExist canon
  145          if exists
  146              then listDir canon
  147              else return ([], [])
  148  
  149      collectTemplates :: [Path Abs File] -> IO [Template]
  150      collectTemplates = fmap catMaybes . mapM loadTemplate <=< filterM isMatch
  151        where
  152          isMatch p = do
  153              ((FP.takeExtension . toFilePath $ p) == ".html" &&)
  154                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)
  155  
  156      findTemplate :: FilePath -> [Path Abs File] -> IO (Maybe Template)
  157      findTemplate name = fmap join . mapM loadTemplate <=< findM isMatch
  158        where
  159          isMatch p =
  160              ((toFilePath . filename $ p) == name &&)
  161                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)

  163  {-
  164  The logic for copying static files and folders. Any file or folder in the
  165  ``confTemplate`` is considered static if:
  166  
  167  - it is a symbolic link, or
  168  - it does not end in ".html" or ".include".
  169  
  170  Symbolic links are not followed and are copied as is. This means that a symbolic link
  171  from `confTemplate/link.html` to `gitserve/index.html` will be copied, keeping the
  172  link intact, resulting in a symbolic link at `output/link.html` essentially
  173  pointing to `output/gitserve/index.html`.
  174  -}
  175  copyStaticDirs :: [Path Abs Dir] -> Path Abs Dir -> IO ()
  176  copyStaticDirs dirs output = mapM_ copy dirs
  177    where
  178      copy :: Path Abs Dir -> IO ()
  179      copy p = do
  180          let isRepo = (toFilePath . dirname $ p) == "repo/"
  181          isLink <- pathIsSymbolicLink . toFilePath $ p
  182          when (not isRepo || isLink) $ do
  183              let output' = output </> dirname p
  184              let fp = FP.dropTrailingPathSeparator . toFilePath $ p
  185              if isLink
  186                  then do
  187                      target <- getSymbolicLinkTarget fp
  188                      createDirectoryLink target . FP.dropTrailingPathSeparator . toFilePath $ output'
  189                  else copyDirRecur p output'
  190  
  191  copyStaticFiles :: [Path Abs File] -> Path Abs Dir -> IO ()
  192  copyStaticFiles files output = mapM_ copy files
  193    where
  194      copy :: Path Abs File -> IO ()
  195      copy p = do
  196          let fp = toFilePath p
  197          let isTemplate = FP.takeExtension fp `elem` [".html", ".include"]
  198          isLink <- pathIsSymbolicLink fp
  199          when (not isTemplate || isLink) $ do
  200              let output' = output </> filename p
  201              if isLink
  202                  then do
  203                      target <- getSymbolicLinkTarget fp
  204                      maybeExists <- forgivingAbsence . isSymlink $ output'
  205                      let exists = fromMaybe False maybeExists
  206                      when exists . removeFile . toFilePath $ output'
  207                      createFileLink target . toFilePath $ output'
  208                  else copyFile p output'
