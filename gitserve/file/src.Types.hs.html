   76  {-
   77  GVal implementation for `Git.Commit r`, allowing commits to be rendered in Ginger
   78  templates.
   79  -}
   80  instance ToGVal m (Commit LgRepo) where
   81      toGVal :: Commit LgRepo -> GVal m
   82      toGVal commit =
   83          def
   84              { asHtml = html . strip . T.takeWhile (/= '\n') . commitLog $ commit
   85              , asText = pack . show . commitLog $ commit
   86              , asLookup = Just . commitAsLookup $ commit
   87              }

   89  commitAsLookup :: Commit LgRepo -> Text -> Maybe (GVal m)
   90  commitAsLookup commit = \case
   91      "id" -> Just . toGVal . show . untag . commitOid $ commit
   92      "href" -> Just . toGVal . (<> ".html") . show . untag . commitOid $ commit
   93      "title" -> Just . toGVal . strip . T.takeWhile (/= '\n') . commitLog $ commit
   94      "body" -> Just . toGVal . strip . T.dropWhile (/= '\n') . commitLog $ commit
   95      "message" -> Just . toGVal . strip . commitLog $ commit
   96      "author" -> Just . toGVal . strip . signatureName . commitAuthor $ commit
   97      "committer" -> Just . toGVal . strip . signatureName . commitCommitter $ commit
   98      "author_email" -> Just . toGVal . strip . signatureEmail . commitAuthor $ commit
   99      "committer_email" -> Just . toGVal . strip . signatureEmail . commitCommitter $ commit
  100      "authored" -> Just . toGVal . show . signatureWhen . commitAuthor $ commit
  101      "committed" -> Just . toGVal . show . signatureWhen . commitCommitter $ commit
  102      "encoding" -> Just . toGVal . strip . commitEncoding $ commit
  103      "parent" -> toGVal . show . untag <$> (listToMaybe . commitParents $ commit)
  104      _ -> Nothing

  121  {-
  122  Some helper functions to convert from Haskell's LibGit2 BlobKind to our TreeEntryMode,

  131  modeToOctal :: TreeEntryMode -> String

  133  modeToOctal ModePlain = "00644"

  135  modeToOctal ModeSymlink = "20000"

  138  modeToSymbolic :: TreeEntryMode -> String
  139  modeToSymbolic ModeDirectory = "drwxr-xr-x"
  140  modeToSymbolic ModePlain = "-rw-r--r--"
  141  modeToSymbolic ModeExecutable = "-rwxr-xr-x"

  145  {-
  146  This
  147  -}
  148  getBlobContents :: BlobOid LgRepo -> ReaderT LgRepo IO TreeFileContents
  149  getBlobContents oid = do
  150      repo <- getRepository
  151      blobPtr <- liftIO mallocForeignPtr
  152      isBinary <- liftIO . withForeignPtr (repoObj repo) $ \repoPtr ->
  153          withForeignPtr blobPtr $ \blobPtr' ->
  154              withForeignPtr (getOid . untag $ oid) $ \oidPtr -> do
  155                  r1 <- c'git_blob_lookup blobPtr' repoPtr oidPtr
  156                  when (r1 < 0) $ throwM (Git.BackendError "Could not lookup blob")
  157                  c'git_blob_is_binary =<< peek blobPtr'
  158  
  159      if toEnum . fromEnum $ isBinary -- This reads weird, but it goes CInt, to Int, to Bool.
  160          then return BinaryContents
  161          else FileContents . decodeUtf8With lenientDecode <$> catBlob oid

  163  {-
  164  GVal implementations for data definitions above, allowing commits to be rendered in

  167  instance ToGVal RunRepo TreeFile where
  168      toGVal :: TreeFile -> GVal RunRepo
  169      toGVal treefile =
  170          def
  171              { asHtml = html . pack . toString . treeFilePath $ treefile
  172              , asText = pack . show . treeFilePath $ treefile
  173              , asLookup = Just . treeAsLookup $ treefile
  174              , asBoolean = True -- Used for conditionally checking readme/license template variables.
  175              }

  177  instance ToGVal RunRepo TreeFileContents where
  178      toGVal :: TreeFileContents -> GVal RunRepo
  179      toGVal BinaryContents = def
  180      toGVal (FileContents text) = toGVal . strip $ text
  181      toGVal (FolderContents treeFiles) =
  182          def
  183              { asHtml = html . pack . show . fmap (toString . treeFilePath) $ treeFiles
  184              , asText = pack . show . fmap (toString . treeFilePath) $ treeFiles
  185              , asList = Just . fmap toGVal $ treeFiles
  186              }

  188  treeAsLookup :: TreeFile -> Text -> Maybe (GVal RunRepo)
  189  treeAsLookup treefile = \case
  190      "path" -> Just . toGVal . treeFilePath $ treefile
  191      "name" -> Just . toGVal . takeFileName . toString . treeFilePath $ treefile
  192      "href" -> Just . toGVal . treePathToHref . treeFilePath $ treefile
  193      "contents" -> Just . toGVal . treeFileContents $ treefile
  194      "mode" -> Just . toGVal . drop 4 . show . treeFileMode $ treefile
  195      "mode_octal" -> Just . toGVal . modeToOctal . treeFileMode $ treefile
  196      "mode_symbolic" -> Just . toGVal . modeToSymbolic . treeFileMode $ treefile
  197      "is_binary" -> Just . toGVal . treeFileIsBinary $ treefile
  198      "is_directory" -> Just . toGVal . treeFileIsDirectory $ treefile
  199      _ -> Nothing
  200  
  201  treeFileIsBinary :: TreeFile -> Bool
  202  treeFileIsBinary treefile = case treeFileContents treefile of
  203      BinaryContents -> True
  204      _ -> False
  205  
  206  treeFileIsDirectory :: TreeFile -> Bool
  207  treeFileIsDirectory treefile = case treeFileContents treefile of
  208      FolderContents _ -> True
  209      _ -> False
