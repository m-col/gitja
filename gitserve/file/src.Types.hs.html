   83      { commitGit :: Git.Commit LgRepo
   84      , commitDiffs :: [Diff]

   87  instance ToGVal m Commit where
   88      toGVal :: Commit -> GVal m
   89      toGVal commit =
   90          def
   91              { asHtml = html . strip . T.takeWhile (/= '\n') . Git.commitLog . commitGit $ commit
   92              , asText = pack . show . Git.commitLog . commitGit $ commit
   93              , asLookup = Just . commitAsLookup $ commit
   94              }

   96  commitAsLookup :: Commit -> Text -> Maybe (GVal m)
   97  commitAsLookup commit = \case
   98      "id" -> Just . toGVal . show . untag . Git.commitOid . commitGit $ commit
   99      "href" -> Just . toGVal . (<> ".html") . show . untag . Git.commitOid . commitGit $ commit
  100      "title" -> Just . toGVal . strip . T.takeWhile (/= '\n') . Git.commitLog . commitGit $ commit
  101      "body" -> Just . toGVal . strip . T.dropWhile (/= '\n') . Git.commitLog . commitGit $ commit
  102      "message" -> Just . toGVal . strip . Git.commitLog . commitGit $ commit
  103      "author" -> Just . toGVal . strip . Git.signatureName . Git.commitAuthor . commitGit $ commit
  104      "committer" -> Just . toGVal . strip . Git.signatureName . Git.commitCommitter . commitGit $ commit
  105      "author_email" -> Just . toGVal . strip . Git.signatureEmail . Git.commitAuthor . commitGit $ commit
  106      "committer_email" -> Just . toGVal . strip . Git.signatureEmail . Git.commitCommitter . commitGit $ commit
  107      "authored" -> Just . toGVal . show . Git.signatureWhen . Git.commitAuthor . commitGit $ commit
  108      "committed" -> Just . toGVal . show . Git.signatureWhen . Git.commitCommitter . commitGit $ commit
  109      "encoding" -> Just . toGVal . strip . Git.commitEncoding . commitGit $ commit
  110      "parent" -> toGVal . show . untag <$> (listToMaybe . Git.commitParents . commitGit $ commit)
  111      "diffs" -> Just . toGVal . commitDiffs $ commit
  112      _ -> Nothing

  114  {-
  115  With `Diff`s there is a hierarchy:
  116  
  117  - A commit has multiple diffs - one per file that changed.
  118  - Each diff has 1 or more hunks
  119  - Each hunk has a number of lines
  120  -}
  121  data Diff = Diff
  122      { diffNewFile :: Git.TreeFilePath
  123      , diffOldFile :: Maybe Git.TreeFilePath
  124      , diffStatus :: Delta
  125      , diffHunks :: [Hunk]
  126      }
  127  
  128  data Delta
  129      = Unmodified
  130      | Added
  131      | Deleted
  132      | Modified
  133      | Renamed
  134      | Copied
  135      | Ignored
  136      | Untracked
  137      | Typechange
  138      deriving stock (Eq, Ord, Enum, Show)
  139  
  140  type HunkHeader = ByteString
  141  type DiffLine = ByteString
  142  
  143  data Hunk = Hunk
  144      { hunkHeader :: HunkHeader
  145      , hunkLines :: [DiffLine]
  146      }
  147  
  148  instance ToGVal m Diff where
  149      toGVal :: Diff -> GVal m
  150      toGVal diff =
  151          def
  152              { asHtml = html . bsToText . diffNewFile $ diff
  153              , asText = bsToText . diffNewFile $ diff
  154              , asLookup = Just . diffAsLookup $ diff
  155              }
  156  
  157  diffAsLookup :: Diff -> Text -> Maybe (GVal m)
  158  diffAsLookup diff = \case
  159      "new_file" -> Just . toGVal . diffNewFile $ diff
  160      "old_file" -> toGVal <$> diffOldFile diff
  161      "status" -> Just . toGVal . show . diffStatus $ diff
  162      "hunks" -> Just . toGVal . diffHunks $ diff
  163      _ -> Nothing
  164  
  165  instance ToGVal m Hunk where
  166      toGVal :: Hunk -> GVal m
  167      toGVal hunk =
  168          def
  169              { asHtml = html . bsToText . hunkHeader $ hunk
  170              , asText = bsToText . hunkHeader $ hunk
  171              , asLookup = Just . hunkAsLookup $ hunk
  172              }
  173  
  174  hunkAsLookup :: Hunk -> Text -> Maybe (GVal m)
  175  hunkAsLookup hunk = \case
  176      "lines" -> Just . toGVal . fmap makeLine . hunkLines $ hunk
  177      "header" -> Just . toGVal . bsToText . hunkHeader $ hunk
  178      _ -> Nothing
  179    where
  180      makeLine :: DiffLine -> Line
  181      makeLine line =
  182          let text = bsToText line
  183           in Line text (cls . T.head $ text)
  184  
  185      cls :: Char -> String
  186      cls = \case
  187          '+' -> "add"
  188          '-' -> "sub"
  189          _ -> "def"

  231  {-
  232  Some helper functions to convert from Haskell's LibGit2 BlobKind to our TreeEntryMode,
  233  and then from that to Git's octal representation, as seen when calling `git ls-tree
  234  <tree-ish>`
  235  -}
  236  blobkindToMode :: Git.BlobKind -> TreeEntryMode
  237  blobkindToMode Git.PlainBlob = ModePlain
  238  blobkindToMode Git.ExecutableBlob = ModeExecutable
  239  blobkindToMode Git.SymlinkBlob = ModeSymlink

  241  {-
  242  This
  243  -}
  244  getBlobContents :: Git.BlobOid LgRepo -> ReaderT LgRepo IO TreeFileContents
  245  getBlobContents oid = do
  246      repo <- Git.getRepository
  247      blobPtr <- liftIO mallocForeignPtr
  248      isBinary <- liftIO . withForeignPtr (repoObj repo) $ \repoPtr ->
  249          withForeignPtr blobPtr $ \blobPtr' ->
  250              withForeignPtr (getOid . untag $ oid) $ \oidPtr -> do
  251                  r1 <- c'git_blob_lookup blobPtr' repoPtr oidPtr
  252                  when (r1 < 0) $ throwM (Git.BackendError "Could not lookup blob")
  253                  c'git_blob_is_binary =<< peek blobPtr'
  254  
  255      if toEnum . fromEnum $ isBinary -- This reads weird, but it goes CInt, to Int, to Bool.
  256          then return BinaryContents
  257          else FileContents <$> Git.catBlob oid

  273  instance ToGVal RunRepo TreeFileContents where
  274      toGVal :: TreeFileContents -> GVal RunRepo
  275      toGVal BinaryContents = def
  276      toGVal (FileContents bytestring) = toGVal bytestring
  277      toGVal (FolderContents treeFiles) =
  278          def
  279              { asHtml = html . pack . show . fmap treeFilePath $ treeFiles
  280              , asText = pack . show . fmap treeFilePath $ treeFiles
  281              , asList = Just . fmap toGVal $ treeFiles
  282              }

  284  treeAsLookup :: TreeFile -> Text -> Maybe (GVal RunRepo)
  285  treeAsLookup treefile = \case
  286      "path" -> Just . toGVal . treeFilePath $ treefile
  287      "name" -> Just . toGVal . FP.takeFileName . T.unpack . treeFilePath $ treefile
  288      "href" -> Just . toGVal . treePathToHref $ treefile
  289      "contents" -> Just . toGVal . treeFileContents $ treefile
  290      "tree" -> Just . toGVal . treeFileGetTree (treeFilePath treefile) . treeFileContents $ treefile
  291      "tree_recursive" -> Just . toGVal . treeFileGetTreeRecursive . treeFileContents $ treefile
  292      "mode" -> Just . toGVal . drop 4 . show . treeFileMode $ treefile
  293      "mode_octal" -> Just . toGVal . modeToOctal . treeFileMode $ treefile
  294      "mode_symbolic" -> Just . toGVal . modeToSymbolic . treeFileMode $ treefile
  295      "is_binary" -> Just . toGVal . treeFileIsBinary $ treefile
  296      "is_directory" -> Just . toGVal . treeFileIsDirectory $ treefile
  297      _ -> Nothing
  298    where
  299      treeFileGetTree :: Text -> TreeFileContents -> [TreeFile]
  300      treeFileGetTree parent (FolderContents fs) = filter atTop fs
  301        where
  302          atTop = notElem FP.pathSeparator . drop 1 . T.unpack . fromMaybe "" . T.stripPrefix parent . treeFilePath
  303      treeFileGetTree _ _ = []
  304  
  305      treeFileGetTreeRecursive :: TreeFileContents -> [TreeFile]
  306      treeFileGetTreeRecursive (FolderContents fs) = fs
  307      treeFileGetTreeRecursive _ = []
  308  
  309      treeFileIsBinary :: TreeFile -> Bool
  310      treeFileIsBinary treefile' = case treeFileContents treefile' of
  311          BinaryContents -> True
  312          _ -> False
  313  
  314      treeFileIsDirectory :: TreeFile -> Bool
  315      treeFileIsDirectory treefile' = case treeFileContents treefile' of
  316          FolderContents _ -> True
  317          _ -> False

  321      modeToOctal ModePlain = "00644"
  322      modeToOctal ModeExecutable = "00755"
  323      modeToOctal ModeSymlink = "20000"

  326      modeToSymbolic :: TreeEntryMode -> String
  327      modeToSymbolic ModeDirectory = "drwxr-xr-x"
  328      modeToSymbolic ModePlain = "-rw-r--r--"
  329      modeToSymbolic ModeExecutable = "-rwxr-xr-x"
