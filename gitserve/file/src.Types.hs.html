   35  {-
   36  GVal implementation for a repository, accessed in the scope of an individual repository,
   37  as well as in a list of all repositories in the index template.
   38  -}
   39  data Repo = Repo
   40      { repositoryPath :: Path Abs Dir
   41      , repositoryDescription :: Text
   42      , repositoryHead :: Maybe (Commit LgRepo)
   43      }

   45  instance ToGVal m (Path b t) where
   46      toGVal :: Path b t -> GVal m
   47      toGVal = toGVal . toFilePath

   49  instance ToGVal m Repo where
   50      toGVal :: Repo -> GVal m
   51      toGVal repo =
   52          def
   53              { asHtml = html . pack . init . unquote . show . dirname . repositoryPath $ repo
   54              , asText = pack . show . dirname . repositoryPath $ repo
   55              , asLookup = Just . repoAsLookup $ repo
   56              }
   57  
   58  unquote :: String -> String
   59  unquote = init . tail
   60  
   61  repoAsLookup :: Repo -> Text -> Maybe (GVal m)
   62  repoAsLookup repo = \case
   63      "name" -> Just . toGVal . init . unquote . show . dirname . repositoryPath $ repo
   64      "description" -> Just . toGVal . repositoryDescription $ repo
   65      "head" -> Just . toGVal . repositoryHead $ repo
   66      "updated" -> toGVal . show . signatureWhen . commitCommitter <$> repositoryHead repo
   67      _ -> Nothing

   69  {-
   70  GVal implementation for `Git.Commit r`, allowing commits to be rendered in Ginger
   71  templates.
   72  -}
   73  instance ToGVal m (Commit LgRepo) where
   74      toGVal :: Commit LgRepo -> GVal m
   75      toGVal commit =
   76          def
   77              { asHtml = html . strip . T.takeWhile (/= '\n') . commitLog $ commit
   78              , asText = pack . show . commitLog $ commit
   79              , asLookup = Just . commitAsLookup $ commit
   80              }
   81  
   82  commitAsLookup :: Commit LgRepo -> Text -> Maybe (GVal m)
   83  commitAsLookup commit = \case
   84      "id" -> Just . toGVal . show . untag . commitOid $ commit
   85      "title" -> Just . toGVal . strip . T.takeWhile (/= '\n') . commitLog $ commit
   86      "body" -> Just . toGVal . strip . T.dropWhile (/= '\n') . commitLog $ commit
   87      "message" -> Just . toGVal . strip . commitLog $ commit
   88      "author" -> Just . toGVal . strip . signatureName . commitAuthor $ commit
   89      "committer" -> Just . toGVal . strip . signatureName . commitCommitter $ commit
   90      "author_email" -> Just . toGVal . strip . signatureEmail . commitAuthor $ commit
   91      "committer_email" -> Just . toGVal . strip . signatureEmail . commitCommitter $ commit
   92      "authored" -> Just . toGVal . show . signatureWhen . commitAuthor $ commit
   93      "committed" -> Just . toGVal . show . signatureWhen . commitCommitter $ commit
   94      "encoding" -> Just . toGVal . strip . commitEncoding $ commit
   95      "parent" -> toGVal . show . untag <$> (listToMaybe . commitParents $ commit)
   96      _ -> Nothing

   98  {-
   99  Next we have some data used to represent a repository's tree and the different kinds of

  102  data TreeFile = TreeFile
  103      { treeFilePath :: TreeFilePath
  104      , treeFileContents :: TreeFileContents
  105      , treeFileMode :: TreeEntryMode
  106      }

  108  data TreeFileContents = FileContents Text | FolderContents [TreeFile]

  110  data TreeEntryMode = ModeDirectory | ModePlain | ModeExecutable | ModeSymlink | ModeSubmodule
  111      deriving stock (Show)
  112  
  113  {-
  114  Some helper functions to convert from Haskell's LibGit2 BlobKind to our TreeEntryMode,
  115  and then from that to Git's octal representation, as seen when calling `git ls-tree
  116  <tree-ish>`
  117  -}
  118  blobkindToMode :: BlobKind -> TreeEntryMode
  119  blobkindToMode PlainBlob = ModePlain
  120  blobkindToMode ExecutableBlob = ModeExecutable
  121  blobkindToMode SymlinkBlob = ModeSymlink
  122  
  123  modeToOctal :: TreeEntryMode -> String
  124  modeToOctal ModeDirectory = "40000"
  125  modeToOctal ModePlain = "00644"
  126  modeToOctal ModeExecutable = "00755"
  127  modeToOctal ModeSymlink = "20000"
  128  modeToOctal ModeSubmodule = "60000"
  129  
  130  modeToSymbolic :: TreeEntryMode -> String
  131  modeToSymbolic ModeDirectory = "drwxr-xr-x"
  132  modeToSymbolic ModePlain = "-rw-r--r--"
  133  modeToSymbolic ModeExecutable = "-rwxr-xr-x"
  134  modeToSymbolic ModeSymlink = "l---------"
  135  modeToSymbolic ModeSubmodule = "git-module"

  137  {-
  138  GVal implementations for data definitions above, allowing commits to be rendered in
  139  Ginger templates.
  140  -}
  141  instance ToGVal RunRepo TreeFile where
  142      toGVal :: TreeFile -> GVal RunRepo
  143      toGVal treefile =
  144          def
  145              { asHtml = html . pack . toString . treeFilePath $ treefile
  146              , asText = pack . show . treeFilePath $ treefile
  147              , asLookup = Just . treeAsLookup $ treefile
  148              , asBoolean = True -- Used for conditionally checking readme/license template variables.
  149              }
  150  
  151  instance ToGVal RunRepo TreeFileContents where
  152      toGVal :: TreeFileContents -> GVal RunRepo
  153      toGVal (FileContents text) = toGVal . strip $ text
  154      toGVal (FolderContents treeFiles) =
  155          def
  156              { asHtml = html . pack . show . fmap (toString . treeFilePath) $ treeFiles
  157              , asText = pack . show . fmap (toString . treeFilePath) $ treeFiles
  158              , asList = Just . fmap toGVal $ treeFiles
  159              }
  160  
  161  treeAsLookup :: TreeFile -> Text -> Maybe (GVal RunRepo)
  162  treeAsLookup treefile = \case
  163      "path" -> Just . toGVal . treeFilePath $ treefile
  164      "href" -> Just . toGVal . treePathToHref . treeFilePath $ treefile
  165      "contents" -> Just . toGVal . treeFileContents $ treefile
  166      "mode" -> Just . toGVal . drop 4 . show . treeFileMode $ treefile
  167      "mode_octal" -> Just . toGVal . modeToOctal . treeFileMode $ treefile
  168      "mode_symbolic" -> Just . toGVal . modeToSymbolic . treeFileMode $ treefile
  169      "is_directory" -> Just . toGVal . treeFileIsDirectory $ treefile
  170      _ -> Nothing
  171  
  172  treeFileIsDirectory :: TreeFile -> Bool
  173  treeFileIsDirectory treefile = case treeFileContents treefile of
  174      FileContents _ -> False
  175      FolderContents _ -> True
  176  
  177  {-
  178  Get the name of a tree file path's HTML file.
  179  -}
  180  treePathToHref :: TreeFilePath -> Text
  181  treePathToHref = flip T.append ".html" . T.replace "/" "." . decodeUtf8With lenientDecode

  183  {-

  191  instance ToGVal RunRepo Ref where
  192      toGVal :: Ref -> GVal RunRepo
  193      toGVal ref =
  194          def
  195              { asHtml = html . refName $ ref
  196              , asText = refName ref
  197              , asLookup = Just . refAsLookup $ ref
  198              }

  200  refAsLookup :: Ref -> Text -> Maybe (GVal RunRepo)
  201  refAsLookup ref = \case
  202      "name" -> Just . toGVal . refName $ ref
  203      "commit" -> Just . toGVal . refCommit $ ref
  204      key -> commitAsLookup (refCommit ref) key
