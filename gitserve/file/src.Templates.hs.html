   30  data Template = Template
   31      { templatePath :: Path Rel File
   32      , templateGinger :: G.Template SourcePos
   33      }
   34  
   35  {-
   36  This tries to load a `Template` from the given file path.
   37  -}
   38  loadTemplate :: Path Abs File -> IO (Maybe Template)
   39  loadTemplate path =
   40      parseGingerFile includeResolver (toFilePath path) >>= \case
   41          Right parsed -> return . Just . Template (filename path) $ parsed
   42          Left err -> do
   43              informError (toFilePath path) err
   44              return Nothing
   45    where
   46      -- An attempt at pretty printing the error message.
   47      informError p (ParserError msg Nothing) =
   48          putStr $ "Template error: " <> p <> "\n" <> indent msg
   49      informError p (ParserError msg (Just pos)) =
   50          putStrLn $ "Template error: " <> p <> "\n" <> indent (show pos <> "\n" <> msg)
   51      indent = unlines . map (mappend "    ") . lines
   52  
   53      -- This resolves template 'includes'.
   54      includeResolver :: FilePath -> IO (Maybe String)
   55      includeResolver p = either (const Nothing) Just <$> tryIOError (readFile p)
   56  
   57  {-
   58  This is the generator function that receives repository-specific variables and uses
   59  Ginger to render templates using them.
   60  -}
   61  generate ::
   62      Path Abs File ->
   63      HashMap.HashMap T.Text (GVal RunRepo) ->
   64      Template ->
   65      ReaderT LgRepo IO ()
   66  generate output context template = do
   67      let output' = toFilePath output
   68      liftIO . ignoringAbsence . removeFile $ output'
   69      content <- liftIO . newIORef . TB.fromText $ ""
   70  
   71      let emit :: Html -> ReaderT LgRepo IO ()
   72          emit = liftIO . modifyIORef' content . flip mappend . TB.fromText . htmlSource
   73  
   74      runGingerT (easyContext emit context) . templateGinger $ template
   75      result <- liftIO . readIORef $ content
   76      liftIO . T.writeFile output' . TB.toLazyText $ result
