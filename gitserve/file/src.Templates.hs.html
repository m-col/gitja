   74  {-
   75  This creates the runtime environment, collecting the config and loading template data
   76  from the template directory.
   77  -}
   78  loadEnv :: Bool -> Bool -> Config -> IO Env
   79  loadEnv quiet force config = do
   80      -- First ensure that the output directory exists
   81      output <- parseAbsDir <=< canonicalizePath . outputDirectory $ config
   82      ensureDir output
   83  
   84      -- Parse repos for env
   85      repoPaths' <-
   86          if scanRepoPaths config
   87              then do
   88                  ps <- fmap concat . mapM (fmap fst . ls) . repoPaths $ config
   89                  return . filter ((/=) ".git" . toFilePath . dirname) $ ps
   90              else mapM (parseAbsDir <=< canonicalizePath) . repoPaths $ config
   91  
   92      -- Find template files, copying the static files as is
   93      (dirs, files) <- ls . templateDirectory $ config
   94      (_, filesRepo) <- ls $ templateDirectory config FP.</> "repo"
   95      copyStaticDirs output dirs
   96      copyStaticFiles output files
   97  
   98      -- Load files from template directory
   99      indexT <- collectTemplates files
  100      commitT <- findTemplate "commit.html" filesRepo
  101      fileT <- findTemplate "file.html" filesRepo
  102      repoT <-
  103          collectTemplates
  104              . filter (flip notElem ["commit.html", "file.html"] . toFilePath . filename)
  105              $ filesRepo
  106  
  107      -- App environment
  108      return
  109          Env
  110              { envConfig = config
  111              , envIndexTemplates = indexT
  112              , envCommitTemplate = commitT
  113              , envFileTemplate = fileT
  114              , envRepoTemplates = repoT
  115              , envOutputDirectory = output
  116              , envRepoPaths = repoPaths'
  117              , envHost = host config
  118              , envQuiet = quiet
  119              , envForce = force
  120              }
  121    where
  122      ls :: FilePath -> IO ([Path Abs Dir], [Path Abs File])
  123      ls dir = do
  124          canon <- parseAbsDir =<< canonicalizePath dir
  125          exists <- doesDirExist canon
  126          if exists
  127              then listDir canon
  128              else return ([], [])
  129  
  130      collectTemplates :: [Path Abs File] -> IO [Template]
  131      collectTemplates = fmap catMaybes . mapM loadTemplate <=< filterM isMatch
  132        where
  133          isMatch p = do
  134              ((FP.takeExtension . toFilePath $ p) == ".html" &&)
  135                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)
  136  
  137      findTemplate :: FilePath -> [Path Abs File] -> IO (Maybe Template)
  138      findTemplate name = fmap join . mapM loadTemplate <=< findM isMatch
  139        where
  140          isMatch p =
  141              ((toFilePath . filename $ p) == name &&)
  142                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)

  144  {-
  145  This is the generator function that receives repository-specific variables and uses
  146  Ginger to render templates using them.
  147  -}
  148  generate ::
  149      Path Abs File ->
  150      HashMap.HashMap T.Text (GVal RunRepo) ->
  151      Template ->
  152      ReaderT LgRepo IO ()
  153  generate output context template = do
  154      let output' = toFilePath output
  155      liftIO . ignoringAbsence . removeFile $ output'
  156      content <- liftIO . newIORef . TB.fromText $ ""
  157  
  158      let emit :: Html -> ReaderT LgRepo IO ()
  159          emit = liftIO . modifyIORef' content . flip mappend . TB.fromText . htmlSource
  160  
  161      runGingerT (easyContext emit context) . templateGinger $ template
  162      result <- liftIO . readIORef $ content
  163      liftIO . T.writeFile output' . TB.toLazyText $ result

  165  {-
  166  This takes the session's `Config` and maybe returns a loaded template for the
  167  ``indexTemplate`` setting.
  168  -}
  169  loadTemplate :: Path Abs File -> IO (Maybe Template)
  170  loadTemplate path =
  171      parseGingerFile includeResolver (toFilePath path) >>= \case
  172          Right parsed -> return . Just . Template (filename path) $ parsed
  173          Left err -> do
  174              informError (toFilePath path) err
  175              return Nothing
  176    where
  177      -- An attempt at pretty printing the error message.
  178      informError p (ParserError msg Nothing) =
  179          putStr $ "Template error: " <> p <> "\n" <> indent msg
  180      informError p (ParserError msg (Just pos)) =
  181          putStrLn $ "Template error: " <> p <> "\n" <> indent (show pos <> "\n" <> msg)
  182      indent = unlines . map (mappend "    ") . lines
  183  
  184      -- This resolves template 'includes'.
  185      includeResolver :: FilePath -> IO (Maybe String)
  186      includeResolver p = either (const Nothing) Just <$> tryIOError (readFile p)

  188  {-
  189  The logic for copying static files and folders. Any file or folder in the
  190  ``templateDirectory`` is considered static if:
  191  
  192  - it is a symbolic link, or
  193  - it does not end in ".html" or ".include".
  194  
  195  Symbolic links are not followed and are copied as is. This means that a symbolic link
  196  from `templateDirectory/link.html` to `gitserve/index.html` will be copied, keeping the
  197  link intact, resulting in a symbolic link at `outputDirectory/link.html` essentially
  198  pointing to `outputDirectory/gitserve/index.html`.
  199  -}
  200  -- TODO: merge isStatic and copy so it's just one step
  201  copyStaticDirs :: Path Abs Dir -> [Path Abs Dir] -> IO ()
  202  copyStaticDirs output = mapM_ copy <=< filterM isStatic
  203    where
  204      isStatic :: Path Abs Dir -> IO Bool
  205      isStatic p = do
  206          let isRepo = (toFilePath . dirname $ p) == "repo/"
  207          isLink <- pathIsSymbolicLink . toFilePath $ p
  208          return $ not isRepo || isLink
  209      copy :: Path Abs Dir -> IO ()
  210      copy p = do
  211          let output' = output </> dirname p
  212          let fp = FP.dropTrailingPathSeparator . toFilePath $ p
  213          isLink <- pathIsSymbolicLink fp
  214          if isLink
  215              then do
  216                  target <- getSymbolicLinkTarget fp
  217                  createDirectoryLink target . FP.dropTrailingPathSeparator . toFilePath $ output'
  218              else copyDirRecur p output'
  219  
  220  copyStaticFiles :: Path Abs Dir -> [Path Abs File] -> IO ()
  221  copyStaticFiles output = mapM_ copy <=< filterM isStatic
  222    where
  223      isStatic :: Path Abs File -> IO Bool
  224      isStatic p = do
  225          let fp = toFilePath p
  226          let isTemplate = FP.takeExtension fp `elem` [".html", ".include"]
  227          isLink <- pathIsSymbolicLink fp
  228          return $ not isTemplate || isLink
  229      copy :: Path Abs File -> IO ()
  230      copy p = do
  231          let fp = toFilePath p
  232          isLink <- pathIsSymbolicLink fp
  233          let output' = output </> filename p
  234          if isLink
  235              then do
  236                  target <- getSymbolicLinkTarget fp
  237                  maybeExists <- forgivingAbsence . isSymlink $ output'
  238                  let exists = fromMaybe False maybeExists
  239                  when exists . removeFile . toFilePath $ output'
  240                  createFileLink target . toFilePath $ output'
  241              else copyFile p output'
