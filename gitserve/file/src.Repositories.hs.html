   36  {-
   37  This is the entrypoint maps over our repositories, reading from them and writing out
   38  their web pages using the loaded templates.
   39  -}
   40  run :: Env -> IO [Repo]
   41  run env = do
   42      repos <- loadRepos env
   43      mapM (processRepo env repos) repos
   44  
   45  {-
   46  Get paths along with their descriptions.
   47  -}
   48  loadRepos :: Env -> IO [Repo]
   49  loadRepos env = do
   50      paths' <- filterM (fmap isRight . okRepo) . envRepos $ env
   51      descs <- mapM getDescription paths'
   52      return . fmap ($ Nothing) . zipWith Repo paths' $ descs
   53    where
   54      okRepo :: Path Abs Dir -> IO (Either GitException LgRepo)
   55      okRepo p = try . liftIO . openRepository lgFactory $ defaultRepositoryOptions{repoPath = toFilePath p}
   56  
   57  {-
   58  Pass the repository's folder, get its description. The algorithm is:
   59  
   60      1. Check for a description in repo/description
   61      2. Check for a description in repo/.git/description
   62      3. Fallback to the repo folder's name.
   63  
   64  -}
   65  getDescription :: Path Abs Dir -> IO Text
   66  getDescription dir =
   67      ifM
   68          (D.doesFileExist inTop)
   69          (return . Just $ inTop)
   70          ( ifM
   71              (D.doesFileExist inGit)
   72              (return . Just $ inGit)
   73              (return Nothing)
   74          )
   75          >>= \case
   76              Just file ->
   77                  strip . pack <$> readFile file
   78              Nothing ->
   79                  return . pack . toFilePath . dirname $ dir
   80    where
   81      inTop = toFilePath dir FP.</> "description"
   82      inGit = toFilePath dir FP.</> ".git" FP.</> "description"
   83  
   84  {-
   85  This receives a file path to a single repository and tries to process it. If the
   86  repository doesn't exist or is unreadable in any way we can forget about it and move on
   87  (after informing the user of course).
   88  -}
   89  processRepo :: Env -> [Repo] -> Repo -> IO Repo
   90  processRepo env repos repo =
   91      withRepository lgFactory (toFilePath . repositoryPath $ repo) $ processRepo' env repos repo
   92  
   93  processRepo' :: Env -> [Repo] -> Repo -> ReaderT LgRepo IO Repo
   94  processRepo' env repos repo = do
   95      let name = dirname . repositoryPath $ repo
   96      let output = envOutput env </> name
   97  
   98      resolveReference "HEAD" >>= \case
   99          Nothing -> do
  100              liftIO . unless (envQuiet env) . putStrLn $ "gitserve: " <> show name <> ": Failed to resolve HEAD."
  101              return repo
  102          Just commitID -> do
  103              let gitHead = Tagged commitID
  104  
  105              -- Collect variables available in the ginger templates --
  106              commits <- getCommits gitHead
  107              tree <- getTree gitHead
  108              tags <- getRefs "refs/tags/"
  109              branches <- getRefs "refs/heads/"
  110              let scope = package env repos name (repositoryDescription repo) commits tree tags branches
  111  
  112              -- Create the destination folders
  113              commitDir <- liftIO . parseRelDir $ "commit"
  114              fileDir <- liftIO . parseRelDir $ "file"
  115              liftIO . ensureDir $ output </> commitDir
  116              liftIO . ensureDir $ output </> fileDir
  117  
  118              -- Run the generator --
  119              let quiet = envQuiet env
  120              let force = envForce env
  121              mapM_ (genRepo output scope) $ envRepoTemplates env
  122              mapM_ (genTarget output scope quiet force $ envCommitTemplate env) commits
  123              mapM_ (genTarget output scope quiet True $ envFileTemplate env) tree -- TODO: detect file changes
  124  
  125              -- Return the repo with the head so the index page can use it. --
  126              return
  127                  repo{repositoryHead = Just . head $ commits}

  129  {-
  130  The role of the function above is to gather information about a git repository and
  131  package it all together in such a way that various parts can be accessed and used by
  132  Ginger templates. `package` takes these pieces of information and places it all into a
  133  hashmap which Ginger can use to look up variables.
  134  -}
  135  package ::
  136      Env ->
  137      [Repo] ->
  138      Path Rel Dir ->
  139      Text ->
  140      [Commit LgRepo] ->
  141      [TreeFile] ->
  142      [Ref] ->
  143      [Ref] ->
  144      HashMap.HashMap Text (GVal RunRepo)
  145  package env repos name description commits tree tags branches =
  146      HashMap.fromList
  147          [ ("host", toGVal . envHost $ env)
  148          , ("repositories", toGVal repos)
  149          , ("name", toGVal . pack . init . toFilePath $ name)
  150          , ("description", toGVal description)
  151          , ("commits", toGVal . reverse $ commits) -- Could be optimised
  152          , ("tree", toGVal tree)
  153          , ("tags", toGVal tags)
  154          , ("branches", toGVal branches)
  155          , ("readme", toGVal . findFile "readme" $ tree)
  156          , ("license", toGVal . findFile "license" $ tree)
  157          ]

  159  {-
  160  Collect commit information.
  161  -}
  162  getCommits :: CommitOid LgRepo -> ReaderT LgRepo IO [Commit LgRepo]
  163  getCommits commitID =
  164      sequence . mapMaybe loadCommit
  165          <=< runConduit
  166          $ sourceObjects Nothing commitID False .| sinkList
  167  
  168  loadCommit :: ObjectOid LgRepo -> Maybe (ReaderT LgRepo IO (Commit LgRepo))
  169  loadCommit (CommitObjOid oid) = Just $ lookupCommit oid
  170  loadCommit _ = Nothing
  171  
  172  {-
  173  Collect tree information for the given commit. Recurses on directories to list their
  174  contents.
  175  -}
  176  getTree :: CommitOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  177  getTree = getTree' "" . commitTree <=< lookupCommit
  178  
  179  getTree' :: TreeFilePath -> TreeOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  180  getTree' parent toid = do
  181      entries <- listTreeEntries =<< lookupTree toid
  182      let entries' = fmap (prependParent parent) entries
  183      contents <- mapM getEntryContents entries'
  184      modes <- mapM (getEntryModes . snd) entries'
  185      return $ zipWith3 TreeFile (fmap fst entries') contents modes
  186  
  187  prependParent :: TreeFilePath -> (TreeFilePath, TreeEntry LgRepo) -> (TreeFilePath, TreeEntry LgRepo)
  188  prependParent "" pathentry = pathentry
  189  prependParent parent (path, entry) = (mconcat [parent, "/", path], entry)
  190  
  191  getEntryContents :: (TreeFilePath, TreeEntry LgRepo) -> ReaderT LgRepo IO TreeFileContents
  192  getEntryContents (_, BlobEntry oid _) = getBlobContents oid
  193  getEntryContents (path, TreeEntry oid) = FolderContents <$> getTree' path oid
  194  getEntryContents (_, CommitEntry oid) = return . FileContents . pack . show . untag $ oid
  195  
  196  getEntryModes :: TreeEntry LgRepo -> ReaderT LgRepo IO TreeEntryMode
  197  getEntryModes (BlobEntry _ kind) = return . blobkindToMode $ kind
  198  getEntryModes (TreeEntry _) = return ModeDirectory
  199  getEntryModes (CommitEntry _) = return ModeSubmodule
  200  
  201  {-
  202  Find a file in the tree starting with the specified prefix. The prefix is looked for on
  203  the full path, so will only find files in the top level directory.
  204  -}
  205  findFile :: Text -> [TreeFile] -> Maybe TreeFile
  206  findFile _ [] = Nothing
  207  findFile prefix (f : fs) = if isReadme f then Just f else findFile prefix fs
  208    where
  209      isReadme = isPrefixOf prefix . toLower . decodeUtf8With lenientDecode . treeFilePath
  210  
  211  {-
  212  Collect information about references. TODO: Find a more canonical way to split
  213  references into tags or branches rather than filtering the refnames.
  214  -}
  215  getRefs :: Text -> ReaderT LgRepo IO [Ref]
  216  getRefs ref = do
  217      names <- filter (isPrefixOf ref) <$> listReferences
  218      maybeOids <- mapM resolveReference names
  219      let names' = catMaybes . zipWith dropName maybeOids $ names
  220      objs <- mapM lookupObject . catMaybes $ maybeOids
  221      maybeCommits <- mapM refObjToCommit objs
  222      let names'' = map (fromJust . stripPrefix ref) . catMaybes . zipWith dropName maybeCommits $ names'
  223      return . zipWith Ref names'' . catMaybes $ maybeCommits
  224    where
  225      refObjToCommit :: Object r (ReaderT LgRepo IO) -> ReaderT LgRepo IO (Maybe (Commit r))
  226      refObjToCommit (CommitObj obj) = return . Just $ obj
  227      refObjToCommit _ = return Nothing
  228  
  229      dropName :: Maybe a -> RefName -> Maybe RefName
  230      dropName (Just _) name = Just name
  231      dropName Nothing _ = Nothing

  244  {-
  245  A Target refers to a template scope and repository object whose information is available
  246  in that scope. For example, commits are a target as they each generate a scope
  247  containing that commit's information, and these scopes are each rendered in the
  248  commitTemplate. The Target class generalises how each target is represented so that
  249  genTarget can work on any target type.
  250  -}
  251  class ToGVal RunRepo a => Target a where
  252      identify :: a -> FilePath
  253      category :: a -> FilePath
  254      dest :: a -> IO (Path Rel File)
  255      dest t = do
  256          dir <- parseRelDir . category $ t
  257          file <- parseRelFile . identify $ t
  258          return $ dir </> file
  259  
  260  instance Target (Commit LgRepo) where
  261      identify = (++ ".html") . show . untag . commitOid
  262      category = const "commit"
  263  
  264  instance Target TreeFile where
  265      identify = unpack . treePathToHref . treeFilePath
  266      category = const "file"
  267  
  268  genTarget ::
  269      Target a =>
  270      Path Abs Dir ->
  271      HashMap.HashMap Text (GVal RunRepo) ->
  272      Bool ->
  273      Bool ->
  274      Maybe Template ->
  275      a ->
  276      ReaderT LgRepo IO ()
  277  genTarget _ _ _ _ Nothing _ = return ()
  278  genTarget output scope quiet force (Just template) target = do
  279      outFile <- liftIO . dest $ target
  280      let output' = output </> outFile
  281      exists <- liftIO . doesFileExist $ output'
  282      when (force || not exists) $ do
  283          liftIO . unless quiet . putStrLn $ "Writing " <> toFilePath output'
  284          let scope' = scope <> HashMap.fromList [(pack . category $ target, toGVal target)]
  285          generate output' scope' template
