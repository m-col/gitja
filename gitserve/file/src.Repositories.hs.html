   34  {-
   35  This is the entrypoint maps over our repositories, reading from them and writing out
   36  their web pages using the loaded templates.
   37  -}
   38  run :: Env -> IO [Repo]
   39  run env = do
   40      repos <- loadRepos env
   41      mapM (processRepo env repos) repos
   42  
   43  {-
   44  Get paths along with their descriptions.
   45  -}
   46  loadRepos :: Env -> IO [Repo]
   47  loadRepos env = do
   48      let paths = envRepoPaths env
   49      descs <- mapM getDescription paths
   50      return . fmap ($ Nothing) . zipWith Repo paths $ descs
   51  
   52  {-
   53  Pass the repository's folder, get its description.
   54  -}
   55  getDescription :: Path Abs Dir -> IO Text
   56  getDescription = fmap (fromRight "") . tryIOError . fmap (strip . pack) . readFile . flip FP.combine "description" . toFilePath
   57  
   58  {-
   59  This receives a file path to a single repository and tries to process it. If the
   60  repository doesn't exist or is unreadable in any way we can forget about it and move on
   61  (after informing the user of course).
   62  -}
   63  processRepo :: Env -> [Repo] -> Repo -> IO Repo
   64  processRepo env repos repo =
   65      withRepository lgFactory (toFilePath . repositoryPath $ repo) $ processRepo' env repos repo
   66  
   67  processRepo' :: Env -> [Repo] -> Repo -> ReaderT LgRepo IO Repo
   68  processRepo' env repos repo = do
   69      let name = dirname . repositoryPath $ repo
   70      let output = envOutputDirectory env </> name
   71  
   72      resolveReference "HEAD" >>= \case
   73          Nothing -> do
   74              liftIO . unless (envQuiet env) . putStrLn $ "gitserve: " <> show name <> ": Failed to resolve HEAD."
   75              return repo
   76          Just commitID -> do
   77              let gitHead = Tagged commitID
   78  
   79              -- Collect variables available in the ginger templates --
   80              commits <- getCommits gitHead
   81              tree <- getTree gitHead
   82              tags <- getRefs "refs/tags/"
   83              branches <- getRefs "refs/heads/"
   84              let scope = package env repos name (repositoryDescription repo) commits tree tags branches
   85  
   86              -- Create the destination folders
   87              commitDir <- liftIO . parseRelDir $ "commit"
   88              fileDir <- liftIO . parseRelDir $ "file"
   89              liftIO . ensureDir $ output </> commitDir
   90              liftIO . ensureDir $ output </> fileDir
   91  
   92              -- Run the generator --
   93              let quiet = envQuiet env
   94              let force = envForce env
   95              mapM_ (genRepo output scope) $ envRepoTemplates env
   96              mapM_ (genTarget output scope quiet force $ envCommitTemplate env) commits
   97              mapM_ (genTarget output scope quiet True $ envFileTemplate env) tree -- TODO: detect file changes
   98  
   99              -- Return the repo with the head so the index page can use it. --
  100              return
  101                  repo{repositoryHead = Just . head $ commits}

  103  {-
  104  The role of the function above is to gather information about a git repository and
  105  package it all together in such a way that various parts can be accessed and used by
  106  Ginger templates. `package` takes these pieces of information and places it all into a
  107  hashmap which Ginger can use to look up variables.
  108  -}
  109  package ::
  110      Env ->
  111      [Repo] ->
  112      Path Rel Dir ->
  113      Text ->
  114      [Commit LgRepo] ->
  115      [TreeFile] ->
  116      [Ref] ->
  117      [Ref] ->
  118      HashMap.HashMap Text (GVal RunRepo)
  119  package env repos name description commits tree tags branches =
  120      HashMap.fromList
  121          [ ("host", toGVal . envHost $ env)
  122          , ("repositories", toGVal repos)
  123          , ("name", toGVal . pack . init . toFilePath $ name)
  124          , ("description", toGVal description)
  125          , ("commits", toGVal . reverse $ commits) -- Could be optimised
  126          , ("tree", toGVal tree)
  127          , ("tags", toGVal tags)
  128          , ("branches", toGVal branches)
  129          , ("readme", toGVal . findFile "readme" $ tree)
  130          , ("license", toGVal . findFile "license" $ tree)
  131          ]

  133  {-
  134  Collect commit information.
  135  -}
  136  getCommits :: CommitOid LgRepo -> ReaderT LgRepo IO [Commit LgRepo]
  137  getCommits commitID =
  138      sequence . mapMaybe loadCommit
  139          <=< runConduit
  140          $ sourceObjects Nothing commitID False .| sinkList
  141  
  142  loadCommit :: ObjectOid LgRepo -> Maybe (ReaderT LgRepo IO (Commit LgRepo))
  143  loadCommit (CommitObjOid oid) = Just $ lookupCommit oid
  144  loadCommit _ = Nothing
  145  
  146  {-
  147  Collect tree information for the given commit. Recurses on directories to list their
  148  contents.
  149  -}
  150  getTree :: CommitOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  151  getTree = getTree' "" . commitTree <=< lookupCommit
  152  
  153  getTree' :: TreeFilePath -> TreeOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  154  getTree' parent toid = do
  155      entries <- listTreeEntries =<< lookupTree toid
  156      let entries' = fmap (prependParent parent) entries
  157      contents <- mapM getEntryContents entries'
  158      modes <- mapM (getEntryModes . snd) entries'
  159      return $ zipWith3 TreeFile (fmap fst entries') contents modes
  160  
  161  prependParent :: TreeFilePath -> (TreeFilePath, TreeEntry LgRepo) -> (TreeFilePath, TreeEntry LgRepo)
  162  prependParent "" pathentry = pathentry
  163  prependParent parent (path, entry) = (mconcat [parent, "/", path], entry)
  164  
  165  getEntryContents :: (TreeFilePath, TreeEntry LgRepo) -> ReaderT LgRepo IO TreeFileContents
  166  getEntryContents (_, BlobEntry oid _) = getBlobContents oid
  167  getEntryContents (path, TreeEntry oid) = FolderContents <$> getTree' path oid
  168  getEntryContents (_, CommitEntry oid) = return . FileContents . pack . show . untag $ oid
  169  
  170  getEntryModes :: TreeEntry LgRepo -> ReaderT LgRepo IO TreeEntryMode
  171  getEntryModes (BlobEntry _ kind) = return . blobkindToMode $ kind
  172  getEntryModes (TreeEntry _) = return ModeDirectory
  173  getEntryModes (CommitEntry _) = return ModeSubmodule
  174  
  175  {-
  176  Find a file in the tree starting with the specified prefix. The prefix is looked for on
  177  the full path, so will only find files in the top level directory.
  178  -}
  179  findFile :: Text -> [TreeFile] -> Maybe TreeFile
  180  findFile _ [] = Nothing
  181  findFile prefix (f : fs) = if isReadme f then Just f else findFile prefix fs
  182    where
  183      isReadme = isPrefixOf prefix . toLower . decodeUtf8With lenientDecode . treeFilePath
  184  
  185  {-
  186  Collect information about references. TODO: Find a more canonical way to split
  187  references into tags or branches rather than filtering the refnames.
  188  -}
  189  getRefs :: Text -> ReaderT LgRepo IO [Ref]
  190  getRefs ref = do
  191      names <- filter (isPrefixOf ref) <$> listReferences
  192      maybeOids <- mapM resolveReference names
  193      let names' = catMaybes . zipWith dropName maybeOids $ names
  194      objs <- mapM lookupObject . catMaybes $ maybeOids
  195      maybeCommits <- mapM refObjToCommit objs
  196      let names'' = map (fromJust . stripPrefix ref) . catMaybes . zipWith dropName maybeCommits $ names'
  197      return . zipWith Ref names'' . catMaybes $ maybeCommits
  198    where
  199      refObjToCommit :: Object r (ReaderT LgRepo IO) -> ReaderT LgRepo IO (Maybe (Commit r))
  200      refObjToCommit (CommitObj obj) = return . Just $ obj
  201      refObjToCommit _ = return Nothing
  202  
  203      dropName :: Maybe a -> RefName -> Maybe RefName
  204      dropName (Just _) name = Just name
  205      dropName Nothing _ = Nothing

  208      Path Abs Dir ->
  209      HashMap.HashMap Text (GVal RunRepo) ->
  210      Template ->
  211      ReaderT LgRepo IO ()

  213      generate (output </> templatePath template) scope template

  218  {-
  219  A Target refers to a template scope and repository object whose information is available
  220  in that scope. For example, commits are a target as they each generate a scope
  221  containing that commit's information, and these scopes are each rendered in the
  222  commitTemplate. The Target class generalises how each target is represented so that
  223  genTarget can work on any target type.
  224  -}
  225  class ToGVal RunRepo a => Target a where
  226      identify :: a -> FilePath
  227      category :: a -> FilePath
  228      dest :: a -> IO (Path Rel File)
  229      dest t = do
  230          dir <- parseRelDir . category $ t
  231          file <- parseRelFile . identify $ t
  232          return $ dir </> file
  233  
  234  instance Target (Commit LgRepo) where
  235      identify = (++ ".html") . show . untag . commitOid
  236      category = const "commit"
  237  
  238  instance Target TreeFile where
  239      identify = unpack . treePathToHref . treeFilePath
  240      category = const "file"
  241  
  242  genTarget ::
  243      Target a =>
  244      Path Abs Dir ->
  245      HashMap.HashMap Text (GVal RunRepo) ->
  246      Bool ->
  247      Bool ->
  248      Maybe Template ->
  249      a ->
  250      ReaderT LgRepo IO ()
  251  genTarget _ _ _ _ Nothing _ = return ()
  252  genTarget output scope quiet force (Just template) target = do
  253      outFile <- liftIO . dest $ target
  254      let output' = output </> outFile
  255      exists <- liftIO . doesFileExist $ output'
  256      when (force || not exists) $ do
  257          liftIO . unless quiet . putStrLn $ "Writing " <> toFilePath output'
  258          let scope' = scope <> HashMap.fromList [(pack . category $ target, toGVal target)]
  259          generate output' scope' template
