   47  {-
   48  This is the entrypoint that maps over our repositories, reading from them and writing
   49  out their web pages using the loaded templates.
   50  -}
   51  run :: Env -> IO [Repo]
   52  run env = do
   53      repos <- loadRepos env
   54      mapM (processRepo env repos) repos
   55  
   56  {-
   57  Get paths along with their descriptions.
   58  -}
   59  loadRepos :: Env -> IO [Repo]
   60  loadRepos = mapM mkRepo <=< filterM (fmap isRight . okRepo) . envRepos
   61    where
   62      okRepo :: Path Abs Dir -> IO (Either Git.GitException LgRepo)
   63      okRepo p =
   64          try . liftIO . Git.openRepository lgFactory $
   65              Git.defaultRepositoryOptions{Git.repoPath = toFilePath p}
   66  
   67      mkRepo :: Path Abs Dir -> IO Repo
   68      mkRepo p = (\d -> Repo p d Nothing) <$> getDescription p
   69  
   70  {-
   71  Pass the repository's folder, get its description. The algorithm is:
   72  
   73      1. Check for a description in repo/description
   74      2. Check for a description in repo/.git/description
   75      3. Fallback to the repo folder's name.
   76  
   77  -}
   78  getDescription :: Path Abs Dir -> IO Text
   79  getDescription dir =
   80      ifM
   81          (D.doesFileExist inTop)
   82          (return . Just $ inTop)
   83          ( ifM
   84              (D.doesFileExist inGit)
   85              (return . Just $ inGit)
   86              (return Nothing)
   87          )
   88          >>= \case
   89              Just file ->
   90                  T.strip . T.pack <$> readFile file
   91              Nothing ->
   92                  return . T.pack . toFilePath . dirname $ dir
   93    where
   94      inTop = toFilePath dir FP.</> "description"
   95      inGit = toFilePath dir FP.</> ".git" FP.</> "description"

   97  {-
   98  This receives a file path to a single repository and tries to process it. If the
   99  repository doesn't exist or is unreadable in any way we can forget about it and move on
  100  (after informing the user of course).
  101  -}
  102  processRepo :: Env -> [Repo] -> Repo -> IO Repo
  103  processRepo env repos repo =
  104      Git.withRepository lgFactory (toFilePath . repositoryPath $ repo) $ processRepo' env repos repo
  105  
  106  processRepo' :: Env -> [Repo] -> Repo -> ReaderT LgRepo IO Repo
  107  processRepo' env repos repo = do
  108      let name = dirname . repositoryPath $ repo
  109      let output = envOutput env </> name
  110  
  111      Git.resolveReference "HEAD" >>= \case
  112          Nothing -> do
  113              liftIO . unless (envQuiet env) . putStrLn $ "gitserve: " <> show name <> ": Failed to resolve HEAD."
  114              return repo
  115          Just commitID -> do
  116              let gitHead = Tagged commitID
  117              headCommit <- loadDiff =<< Git.lookupCommit gitHead
  118  
  119              -- If a page exists for the head commit, don't do anything else --
  120              exists <-
  121                  liftIO . D.doesFileExist $
  122                      toFilePath output FP.</> "commit" FP.</> show commitID <> ".html"
  123  
  124              when (not exists || envForce env) $ do
  125                  -- Collect variables available to the ginger templates --
  126                  commits <- getCommits gitHead
  127                  tree <- getTree gitHead
  128                  tags <- getRefs "refs/tags/"
  129                  branches <- getRefs "refs/heads/"
  130                  let scope = package env repos name (repositoryDescription repo) commits tree tags branches
  131  
  132                  -- Create the destination folders --
  133                  commitDir <- liftIO . parseRelDir $ "commit"
  134                  fileDir <- liftIO . parseRelDir $ "file"
  135                  liftIO . ensureDir $ output </> commitDir
  136                  liftIO . ensureDir $ output </> fileDir
  137  
  138                  -- Run the generator --
  139                  let quiet = envQuiet env
  140                  let force = envForce env
  141                  mapM_ (genRepo output scope) $ envRepoTemplates env
  142                  mapM_ (genTarget output scope quiet force $ envCommitTemplate env) commits
  143                  mapM_ (genTarget output scope quiet True $ envFileTemplate env) tree -- TODO: detect file changes
  144  
  145                  -- Copy any static files/folders into the output folder --
  146                  liftIO . envRepoCopyStatics env $ output
  147  
  148              return
  149                  repo{repositoryHead = Just headCommit}

  151  {-
  152  The role of the function above is to gather information about a git repository and
  153  package it all together in such a way that various parts can be accessed and used by
  154  Ginger templates. `package` takes these pieces of information and places it all into a
  155  hashmap which Ginger can use to look up variables.
  156  -}
  157  package ::
  158      Env ->
  159      [Repo] ->
  160      Path Rel Dir ->
  161      Text ->
  162      [Commit] ->
  163      [TreeFile] ->
  164      [Ref] ->
  165      [Ref] ->
  166      HashMap.HashMap Text (GVal RunRepo)
  167  package env repos name description commits tree tags branches =
  168      HashMap.fromList
  169          [ ("host", toGVal . envHost $ env)
  170          , ("repositories", toGVal repos)
  171          , ("name", toGVal . T.pack . init . toFilePath $ name)
  172          , ("description", toGVal description)
  173          , ("commits", toGVal commits)
  174          , ("tree", toGVal . filter (notElem FP.pathSeparator . T.unpack . treeFilePath) $ tree)
  175          , ("tree_recursive", toGVal tree)
  176          , ("tags", toGVal tags)
  177          , ("branches", toGVal branches)
  178          , ("readme", toGVal . findFile "readme" $ tree)
  179          , ("license", toGVal . findFile "license" $ tree)
  180          ]

  182  {-
  183  Collect commit information.
  184  -}
  185  getCommits :: Git.CommitOid LgRepo -> ReaderT LgRepo IO [Commit]
  186  getCommits commitID =
  187      fmap reverse . sequence . mapMaybe loadCommit
  188          <=< runConduit
  189          $ Git.sourceObjects Nothing commitID False .| sinkList
  190  
  191  loadCommit :: Git.ObjectOid LgRepo -> Maybe (ReaderT LgRepo IO Commit)
  192  loadCommit (Git.CommitObjOid oid) = Just $ loadDiff =<< Git.lookupCommit oid
  193  loadCommit _ = Nothing
  194  
  195  loadDiff :: Git.Commit LgRepo -> ReaderT LgRepo IO Commit
  196  loadDiff gitCommit = do
  197      newTree <- Git.lookupTree . Git.commitTree $ gitCommit
  198      oldTree <-
  199          case listToMaybe . Git.commitParents $ gitCommit of
  200              Just parent ->
  201                  fmap Just . Git.lookupTree . Git.commitTree =<< Git.lookupCommit parent
  202              Nothing ->
  203                  return Nothing
  204  
  205      ioref <- liftIO . newIORef $ []
  206      lgDiffTreeToTree
  207          (fileCallback ioref)
  208          (hunkCallback ioref)
  209          (dataCallback ioref)
  210          oldTree
  211          (Just newTree)
  212      diffs <- liftIO . readIORef $ ioref
  213  
  214      return . Commit gitCommit . fmap fixupLists $ diffs
  215    where
  216      fileCallback ::
  217          IORef [Diff] ->
  218          Ptr LG.C'git_diff_delta ->
  219          CFloat ->
  220          Ptr () ->
  221          IO CInt
  222      fileCallback ioref dPtr _progress _payload = do
  223          delta <- peek dPtr
  224          newFile <- B.packCString . LG.c'git_diff_file'path . LG.c'git_diff_delta'new_file $ delta
  225          oldFile <- B.packCString . LG.c'git_diff_file'path . LG.c'git_diff_delta'old_file $ delta
  226          let oldFile' = bool Nothing (Just oldFile) (newFile /= oldFile)
  227              status = toEnum . fromIntegral . LG.c'git_diff_delta'status $ delta
  228              diff = Diff newFile oldFile' status []
  229          modifyIORef ioref (diff :)
  230          return 0
  231  
  232      hunkCallback ::
  233          IORef [Diff] ->
  234          Ptr LG.C'git_diff_delta ->
  235          Ptr LG.C'git_diff_range ->
  236          CString ->
  237          CSize ->
  238          Ptr () ->
  239          IO CInt
  240      hunkCallback ioref _dPtr _range header headerLen _payload = do
  241          (cur : _, rest) <- splitAt 1 <$> readIORef ioref
  242          bs <- curry B.packCStringLen header (fromIntegral headerLen)
  243          let hunk = Hunk bs []
  244          writeIORef ioref $ cur{diffHunks = hunk : diffHunks cur} : rest
  245          return 0
  246  
  247      dataCallback ::
  248          IORef [Diff] ->
  249          Ptr LG.C'git_diff_delta ->
  250          Ptr LG.C'git_diff_range ->
  251          CChar ->
  252          CString ->
  253          CSize ->
  254          Ptr () ->
  255          IO CInt
  256      dataCallback ioref _dPtr _range lineOrigin content contentLen _payload = do
  257          bs <- curry B.packCStringLen content (fromIntegral contentLen)
  258          let bs' = B.cons (fromIntegral lineOrigin) bs
  259          (cur : _, rest) <- splitAt 1 <$> readIORef ioref
  260          let (curHunk : _, restHunks) = splitAt 1 . diffHunks $ cur
  261          let updated =
  262                  cur
  263                      { diffHunks =
  264                          curHunk
  265                              { hunkLines = bs' : hunkLines curHunk
  266                              } :
  267                          restHunks
  268                      }
  269          writeIORef ioref $ updated : rest
  270          return 0
  271  
  272      -- The callbacks prepend and then we put the lists the right way round here.
  273      -- This avoids traversing the lists every time we add an item.
  274      fixupLists :: Diff -> Diff
  275      fixupLists diff = diff{diffHunks = fmap fixupLines . reverse . diffHunks $ diff}
  276        where
  277          fixupLines hunk = hunk{hunkLines = reverse . hunkLines $ hunk}

  279  {-
  280  Collect tree information for the given commit. Recurses on directories to list their
  281  contents.
  282  -}
  283  getTree :: Git.CommitOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  284  getTree = getTree' "" . Git.commitTree <=< Git.lookupCommit
  285  
  286  getTree' :: Git.TreeFilePath -> Git.TreeOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  287  getTree' parent toid = do
  288      entries <- Git.listTreeEntries =<< Git.lookupTree toid
  289      let entries' = fmap (prependParent parent) entries
  290      contents <- mapM getEntryContents entries'
  291      modes <- mapM (getEntryModes . snd) entries'
  292      return $ zipWith3 TreeFile (fmap treePaths entries') contents modes
  293  
  294  prependParent :: Git.TreeFilePath -> (Git.TreeFilePath, Git.TreeEntry LgRepo) -> (Git.TreeFilePath, Git.TreeEntry LgRepo)
  295  prependParent "" pathentry = pathentry
  296  prependParent parent (path, entry) = (mconcat [parent, "/", path], entry)
  297  
  298  getEntryContents :: (Git.TreeFilePath, Git.TreeEntry LgRepo) -> ReaderT LgRepo IO TreeFileContents
  299  getEntryContents (_, Git.BlobEntry oid _) = getBlobContents oid
  300  getEntryContents (path, Git.TreeEntry oid) = FolderContents <$> getTree' path oid
  301  getEntryContents (_, Git.CommitEntry oid) = return . FileContents . B.fromString . show . untag $ oid
  302  
  303  getEntryModes :: Git.TreeEntry LgRepo -> ReaderT LgRepo IO TreeEntryMode
  304  getEntryModes (Git.BlobEntry _ kind) = return . blobkindToMode $ kind
  305  getEntryModes (Git.TreeEntry _) = return ModeDirectory
  306  getEntryModes (Git.CommitEntry _) = return ModeSubmodule
  307  
  308  treePaths :: (Git.TreeFilePath, Git.TreeEntry LgRepo) -> Text
  309  treePaths = T.decodeUtf8With T.lenientDecode . fst

  311  {-
  312  Find a file in the tree starting with the specified prefix. The prefix is looked for on
  313  the full path, so will only find files in the top level directory.
  314  -}
  315  findFile :: Text -> [TreeFile] -> Maybe TreeFile
  316  findFile _ [] = Nothing
  317  findFile prefix (f : fs) = if isReadme f then Just f else findFile prefix fs
  318    where
  319      isReadme = T.isPrefixOf prefix . T.toLower . treeFilePath

  321  {-
  322  Collect information about references. TODO: Find a more canonical way to split
  323  references into tags or branches rather than filtering the refnames.
  324  -}
  325  getRefs :: Text -> ReaderT LgRepo IO [Ref]
  326  getRefs ref = do
  327      names <- filter (T.isPrefixOf ref) <$> Git.listReferences
  328      maybeOids <- mapM Git.resolveReference names
  329      let names' = catMaybes . zipWith dropName maybeOids $ names
  330      objs <- mapM Git.lookupObject . catMaybes $ maybeOids
  331      maybeCommits <- mapM refObjToCommit objs
  332      let names'' = map (fromJust . T.stripPrefix ref) . catMaybes . zipWith dropName maybeCommits $ names'
  333      return . zipWith Ref names'' . catMaybes $ maybeCommits
  334    where
  335      refObjToCommit ::
  336          Git.Object LgRepo (ReaderT LgRepo IO) ->
  337          ReaderT LgRepo IO (Maybe Commit)
  338      refObjToCommit (Git.CommitObj obj) = Just <$> loadDiff obj
  339      refObjToCommit _ = return Nothing
  340  
  341      dropName :: Maybe a -> Git.RefName -> Maybe Git.RefName
  342      dropName (Just _) name = Just name
  343      dropName Nothing _ = Nothing

  356  {-
  357  A Target refers to a template scope and repository object whose information is available
  358  in that scope. For example, commits are a target as they each generate a scope
  359  containing that commit's information, and these scopes are each rendered in the
  360  commitTemplate. The Target class generalises how each target is represented so that
  361  genTarget can work on any target type.
  362  -}
  363  class ToGVal RunRepo a => Target a where
  364      identify :: a -> FilePath
  365      category :: a -> FilePath
  366      dest :: a -> IO (Path Rel File)
  367      dest t = do
  368          dir <- parseRelDir . category $ t
  369          file <- parseRelFile . identify $ t
  370          return $ dir </> file
  371  
  372  instance Target Commit where
  373      identify = (++ ".html") . show . untag . Git.commitOid . commitGit
  374      category = const "commit"
  375  
  376  instance Target TreeFile where
  377      identify = T.unpack . treePathToHref
  378      category = const "file"
  379  
  380  genTarget ::
  381      Target a =>
  382      Path Abs Dir ->
  383      HashMap.HashMap Text (GVal RunRepo) ->
  384      Bool ->
  385      Bool ->
  386      Maybe Template ->
  387      a ->
  388      ReaderT LgRepo IO ()
  389  genTarget _ _ _ _ Nothing _ = return ()
  390  genTarget output scope quiet force (Just template) target = do
  391      outFile <- liftIO . dest $ target
  392      let output' = output </> outFile
  393      exists <- liftIO . doesFileExist $ output'
  394      when (force || not exists) $ do
  395          liftIO . unless quiet . putStrLn $ "Writing " <> toFilePath output'
  396          let scope' = scope <> HashMap.fromList [(T.pack . category $ target, toGVal target)]
  397          generate output' scope' template
