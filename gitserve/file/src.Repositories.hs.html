   38  {-
   39  This is the entrypoint maps over our repositories, reading from them and writing out
   40  their web pages using the loaded templates.
   41  -}
   42  run :: Env -> IO [Repo]
   43  run env = do
   44      repos <- loadRepos env
   45      mapM (processRepo env repos) repos
   46  
   47  {-
   48  Get paths along with their descriptions.
   49  -}
   50  loadRepos :: Env -> IO [Repo]
   51  loadRepos env = do
   52      paths' <- filterM (fmap isRight . okRepo) . envRepos $ env
   53      descs <- mapM getDescription paths'
   54      return . fmap ($ Nothing) . zipWith Repo paths' $ descs
   55    where
   56      okRepo :: Path Abs Dir -> IO (Either GitException LgRepo)
   57      okRepo p = try . liftIO . openRepository lgFactory $ defaultRepositoryOptions{repoPath = toFilePath p}
   58  
   59  {-
   60  Pass the repository's folder, get its description. The algorithm is:
   61  
   62      1. Check for a description in repo/description
   63      2. Check for a description in repo/.git/description
   64      3. Fallback to the repo folder's name.
   65  
   66  -}
   67  getDescription :: Path Abs Dir -> IO Text
   68  getDescription dir =
   69      ifM
   70          (D.doesFileExist inTop)
   71          (return . Just $ inTop)
   72          ( ifM
   73              (D.doesFileExist inGit)
   74              (return . Just $ inGit)
   75              (return Nothing)
   76          )
   77          >>= \case
   78              Just file ->
   79                  strip . pack <$> readFile file
   80              Nothing ->
   81                  return . pack . toFilePath . dirname $ dir
   82    where
   83      inTop = toFilePath dir FP.</> "description"
   84      inGit = toFilePath dir FP.</> ".git" FP.</> "description"
   85  
   86  {-
   87  This receives a file path to a single repository and tries to process it. If the
   88  repository doesn't exist or is unreadable in any way we can forget about it and move on
   89  (after informing the user of course).
   90  -}
   91  processRepo :: Env -> [Repo] -> Repo -> IO Repo
   92  processRepo env repos repo =
   93      withRepository lgFactory (toFilePath . repositoryPath $ repo) $ processRepo' env repos repo
   94  
   95  processRepo' :: Env -> [Repo] -> Repo -> ReaderT LgRepo IO Repo
   96  processRepo' env repos repo = do
   97      let name = dirname . repositoryPath $ repo
   98      let output = envOutput env </> name
   99  
  100      resolveReference "HEAD" >>= \case
  101          Nothing -> do
  102              liftIO . unless (envQuiet env) . putStrLn $ "gitserve: " <> show name <> ": Failed to resolve HEAD."
  103              return repo
  104          Just commitID -> do
  105              let gitHead = Tagged commitID
  106  
  107              -- Collect variables available in the ginger templates --
  108              commits <- getCommits gitHead
  109              tree <- getTree gitHead
  110              tags <- getRefs "refs/tags/"
  111              branches <- getRefs "refs/heads/"
  112              let scope = package env repos name (repositoryDescription repo) commits tree tags branches
  113  
  114              -- Create the destination folders --
  115              commitDir <- liftIO . parseRelDir $ "commit"
  116              fileDir <- liftIO . parseRelDir $ "file"
  117              liftIO . ensureDir $ output </> commitDir
  118              liftIO . ensureDir $ output </> fileDir
  119  
  120              -- Run the generator --
  121              let quiet = envQuiet env
  122              let force = envForce env
  123              mapM_ (genRepo output scope) $ envRepoTemplates env
  124              mapM_ (genTarget output scope quiet force $ envCommitTemplate env) commits
  125              mapM_ (genTarget output scope quiet True $ envFileTemplate env) tree -- TODO: detect file changes
  126  
  127              -- Copy any static files/folders into the output folder --
  128              liftIO . envRepoCopyStatics env $ output
  129  
  130              -- Return the repo with the head so the index page can use it. --
  131              return
  132                  repo{repositoryHead = Just . head $ commits}
  133  
  134  {-
  135  The role of the function above is to gather information about a git repository and
  136  package it all together in such a way that various parts can be accessed and used by
  137  Ginger templates. `package` takes these pieces of information and places it all into a
  138  hashmap which Ginger can use to look up variables.
  139  -}
  140  package ::
  141      Env ->
  142      [Repo] ->
  143      Path Rel Dir ->
  144      Text ->
  145      [Commit LgRepo] ->
  146      [TreeFile] ->
  147      [Ref] ->
  148      [Ref] ->
  149      HashMap.HashMap Text (GVal RunRepo)
  150  package env repos name description commits tree tags branches =
  151      HashMap.fromList
  152          [ ("host", toGVal . envHost $ env)
  153          , ("repositories", toGVal repos)
  154          , ("name", toGVal . pack . init . toFilePath $ name)
  155          , ("description", toGVal description)
  156          , ("commits", toGVal commits)
  157          , ("tree", toGVal . filter (notElem FP.pathSeparator . toString . treeFilePath) $ tree)
  158          , ("tree_recursive", toGVal tree)
  159          , ("tags", toGVal tags)
  160          , ("branches", toGVal branches)
  161          , ("readme", toGVal . findFile "readme" $ tree)
  162          , ("license", toGVal . findFile "license" $ tree)
  163          ]

  165  {-
  166  Collect commit information.
  167  -}
  168  getCommits :: CommitOid LgRepo -> ReaderT LgRepo IO [Commit LgRepo]
  169  getCommits commitID =
  170      fmap reverse . sequence . mapMaybe loadCommit
  171          <=< runConduit
  172          $ sourceObjects Nothing commitID False .| sinkList
  173  
  174  loadCommit :: ObjectOid LgRepo -> Maybe (ReaderT LgRepo IO (Commit LgRepo))
  175  loadCommit (CommitObjOid oid) = Just $ lookupCommit oid
  176  loadCommit _ = Nothing
  177  
  178  {-
  179  Collect tree information for the given commit. Recurses on directories to list their
  180  contents.
  181  -}
  182  getTree :: CommitOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  183  getTree = getTree' "" . commitTree <=< lookupCommit
  184  
  185  getTree' :: TreeFilePath -> TreeOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  186  getTree' parent toid = do
  187      entries <- listTreeEntries =<< lookupTree toid
  188      let entries' = fmap (prependParent parent) entries
  189      contents <- mapM getEntryContents entries'
  190      modes <- mapM (getEntryModes . snd) entries'
  191      return $ zipWith3 TreeFile (fmap fst entries') contents modes
  192  
  193  prependParent :: TreeFilePath -> (TreeFilePath, TreeEntry LgRepo) -> (TreeFilePath, TreeEntry LgRepo)
  194  prependParent "" pathentry = pathentry
  195  prependParent parent (path, entry) = (mconcat [parent, "/", path], entry)
  196  
  197  getEntryContents :: (TreeFilePath, TreeEntry LgRepo) -> ReaderT LgRepo IO TreeFileContents
  198  getEntryContents (_, BlobEntry oid _) = getBlobContents oid
  199  getEntryContents (path, TreeEntry oid) = FolderContents <$> getTree' path oid
  200  getEntryContents (_, CommitEntry oid) = return . FileContents . pack . show . untag $ oid
  201  
  202  getEntryModes :: TreeEntry LgRepo -> ReaderT LgRepo IO TreeEntryMode
  203  getEntryModes (BlobEntry _ kind) = return . blobkindToMode $ kind
  204  getEntryModes (TreeEntry _) = return ModeDirectory
  205  getEntryModes (CommitEntry _) = return ModeSubmodule
  206  
  207  {-
  208  Find a file in the tree starting with the specified prefix. The prefix is looked for on
  209  the full path, so will only find files in the top level directory.
  210  -}
  211  findFile :: Text -> [TreeFile] -> Maybe TreeFile
  212  findFile _ [] = Nothing
  213  findFile prefix (f : fs) = if isReadme f then Just f else findFile prefix fs
  214    where
  215      isReadme = isPrefixOf prefix . toLower . decodeUtf8With lenientDecode . treeFilePath
  216  
  217  {-
  218  Collect information about references. TODO: Find a more canonical way to split
  219  references into tags or branches rather than filtering the refnames.
  220  -}
  221  getRefs :: Text -> ReaderT LgRepo IO [Ref]
  222  getRefs ref = do
  223      names <- filter (isPrefixOf ref) <$> listReferences
  224      maybeOids <- mapM resolveReference names
  225      let names' = catMaybes . zipWith dropName maybeOids $ names
  226      objs <- mapM lookupObject . catMaybes $ maybeOids
  227      maybeCommits <- mapM refObjToCommit objs
  228      let names'' = map (fromJust . stripPrefix ref) . catMaybes . zipWith dropName maybeCommits $ names'
  229      return . zipWith Ref names'' . catMaybes $ maybeCommits
  230    where
  231      refObjToCommit :: Object r (ReaderT LgRepo IO) -> ReaderT LgRepo IO (Maybe (Commit r))
  232      refObjToCommit (CommitObj obj) = return . Just $ obj
  233      refObjToCommit _ = return Nothing
  234  
  235      dropName :: Maybe a -> RefName -> Maybe RefName
  236      dropName (Just _) name = Just name
  237      dropName Nothing _ = Nothing

  250  {-
  251  A Target refers to a template scope and repository object whose information is available
  252  in that scope. For example, commits are a target as they each generate a scope
  253  containing that commit's information, and these scopes are each rendered in the
  254  commitTemplate. The Target class generalises how each target is represented so that
  255  genTarget can work on any target type.
  256  -}
  257  class ToGVal RunRepo a => Target a where
  258      identify :: a -> FilePath
  259      category :: a -> FilePath
  260      dest :: a -> IO (Path Rel File)
  261      dest t = do
  262          dir <- parseRelDir . category $ t
  263          file <- parseRelFile . identify $ t
  264          return $ dir </> file
  265  
  266  instance Target (Commit LgRepo) where
  267      identify = (++ ".html") . show . untag . commitOid
  268      category = const "commit"
  269  
  270  instance Target TreeFile where
  271      identify = unpack . treePathToHref . treeFilePath
  272      category = const "file"
  273  
  274  genTarget ::
  275      Target a =>
  276      Path Abs Dir ->
  277      HashMap.HashMap Text (GVal RunRepo) ->
  278      Bool ->
  279      Bool ->
  280      Maybe Template ->
  281      a ->
  282      ReaderT LgRepo IO ()
  283  genTarget _ _ _ _ Nothing _ = return ()
  284  genTarget output scope quiet force (Just template) target = do
  285      outFile <- liftIO . dest $ target
  286      let output' = output </> outFile
  287      exists <- liftIO . doesFileExist $ output'
  288      when (force || not exists) $ do
  289          liftIO . unless quiet . putStrLn $ "Writing " <> toFilePath output'
  290          let scope' = scope <> HashMap.fromList [(pack . category $ target, toGVal target)]
  291          generate output' scope' template
