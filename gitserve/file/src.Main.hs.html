    1  {-# LANGUAGE OverloadedStrings #-}
    2  
    3  module Main (
    4      main,
    5  ) where
    6  
    7  import Paths_gitserve (version)
    8  
    9  import Data.Version (showVersion)
   10  import qualified Options.Applicative as O
   11  
   12  import Config (getConfig)
   13  import Index (runIndex)
   14  import Repositories (run)
   15  import Templates (loadEnv)
   16  
   17  {-
   18  Command line options
   19  -}
   20  data Options = Options
   21      { optConfig :: FilePath
   22      , optQuiet :: Bool
   23      , optForce :: Bool
   24      , optVersion :: Bool
   25      }
   26  
   27  opts :: O.Parser Options
   28  opts =
   29      Options
   30          <$> O.strOption
   31              ( O.long "config"
   32                  <> O.short 'c'
   33                  <> O.metavar "CONFIG"
   34                  <> O.value "./config.dhall"
   35                  <> O.help "Configuration file to use (Default: ./config.dhall)."
   36              )
   37          <*> O.switch
   38              ( O.long "quiet"
   39                  <> O.short 'q'
   40                  <> O.help "Suppress non-error output."
   41              )
   42          <*> O.switch
   43              ( O.long "force"
   44                  <> O.short 'f'
   45                  <> O.help "Force regeneration of all files."
   46              )
   47          <*> O.switch
   48              ( O.long "version"
   49                  <> O.short 'v'
   50                  <> O.help "Print the gitserve's version."
   51              )
   52  
   53  parser :: IO Options
   54  parser =
   55      O.execParser $
   56          O.info
   57              (opts O.<**> O.helper)
   58              ( O.progDesc "üêô Templated web page generator for your git repositories"
   59              )
   60  
   61  {-
   62  Main logic
   63  -}
   64  main :: IO ()
   65  main = do
   66      options <- parser
   67      if optVersion options
   68          then putStrLn $ "Your gitserve version is: " <> showVersion version
   69          else do
   70              conf <- getConfig (optConfig options)
   71              env <- loadEnv (optQuiet options) (optForce options) conf
   72              runIndex env =<< run env
