   44  {-
   45  The Config data type represents the configuration options available in the config file.
   46  Each record of the type, without the 'conf' prefix, is an option.
   47  -}
   48  data Config = Config
   49      { confRepos :: [FilePath]
   50      , confScan :: Bool
   51      , confTemplate :: FilePath
   52      , confOutput :: FilePath
   53      , confHost :: Text
   54      }
   55      deriving stock (Generic)
   56      deriving
   57          (FromDhall)
   58          via Codec (Field (CamelCase <<< DropPrefix "conf")) Config

   60  getConfig :: String -> IO Config
   61  getConfig = input auto . T.pack <=< makeAbsolute

   63  {-
   64  The Env data type represents all of the program's state, including user configuration
   65  and loaded template data. This can be accessed as immutable global state at any point.
   66  -}
   67  data Env = Env
   68      { envConfig :: Config
   69      , envIndexTemplates :: [Template]
   70      , envCommitTemplate :: Maybe Template
   71      , envFileTemplate :: Maybe Template
   72      , envRepoTemplates :: [Template]
   73      , envOutput :: Path Abs Dir
   74      , envRepos :: [Path Abs Dir]
   75      , envHost :: T.Text
   76      , envQuiet :: Bool
   77      , envForce :: Bool
   78      , envRepoCopyStatics :: Path Abs Dir -> IO ()
   79      }

   81  {-
   82  This creates the runtime environment, collecting the config and loading template data
   83  from the template directory.
   84  -}
   85  loadEnv :: Bool -> Bool -> Config -> IO Env
   86  loadEnv quiet force config = do
   87      -- First ensure that the output directory exists
   88      output <- parseAbsDir <=< canonicalizePath . confOutput $ config
   89      ensureDir output
   90  
   91      -- Parse repos for env
   92      repos <-
   93          if confScan config
   94              then do
   95                  ps <- fmap concat . mapM (fmap fst . ls) . confRepos $ config
   96                  return . filter ((/=) ".git" . toFilePath . dirname) $ ps
   97              else mapM (parseAbsDir <=< canonicalizePath) . confRepos $ config
   98  
   99      -- Find template files, copying the static files as is
  100      (dirs, files) <- ls . confTemplate $ config
  101      (dirsRepo, filesRepo) <- ls $ confTemplate config FP.</> "repo"
  102      copyStaticDirs dirs output
  103      copyStaticFiles files output
  104  
  105      -- Load files from template directory
  106      indexT <- collectTemplates files
  107      commitT <- findTemplate "commit.html" filesRepo
  108      fileT <- findTemplate "file.html" filesRepo
  109      repoT <-
  110          collectTemplates
  111              . filter (flip notElem ["commit.html", "file.html"] . toFilePath . filename)
  112              $ filesRepo
  113  
  114      -- Exit early if we didn't find any templates
  115      when
  116          ( all null [indexT, repoT] && all isNothing [commitT, fileT]
  117          )
  118          $ die "No templates were found."
  119  
  120      -- App environment
  121      return
  122          Env
  123              { envConfig = config
  124              , envIndexTemplates = indexT
  125              , envCommitTemplate = commitT
  126              , envFileTemplate = fileT
  127              , envRepoTemplates = repoT
  128              , envOutput = output
  129              , envRepos = repos
  130              , envHost = confHost config
  131              , envQuiet = quiet
  132              , envForce = force
  133              , envRepoCopyStatics = \p -> copyStaticDirs dirsRepo p >> copyStaticFiles filesRepo p
  134              }
  135    where
  136      ls :: FilePath -> IO ([Path Abs Dir], [Path Abs File])
  137      ls dir = do
  138          canon <- parseAbsDir =<< canonicalizePath dir
  139          exists <- doesDirExist canon
  140          if exists
  141              then listDir canon
  142              else return ([], [])
  143  
  144      collectTemplates :: [Path Abs File] -> IO [Template]
  145      collectTemplates = fmap catMaybes . mapM loadTemplate <=< filterM isMatch
  146        where
  147          isMatch p = do
  148              ((FP.takeExtension . toFilePath $ p) == ".html" &&)
  149                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)
  150  
  151      findTemplate :: FilePath -> [Path Abs File] -> IO (Maybe Template)
  152      findTemplate name = fmap join . mapM loadTemplate <=< findM isMatch
  153        where
  154          isMatch p =
  155              ((toFilePath . filename $ p) == name &&)
  156                  <$> (fmap not . pathIsSymbolicLink . toFilePath $ p)

  158  {-
  159  The logic for copying static files and folders. Any file or folder in the
  160  ``confTemplate`` is considered static if:
  161  
  162  - it is a symbolic link, or
  163  - it does not end in ".html" or ".include".
  164  
  165  Symbolic links are not followed and are copied as is. This means that a symbolic link
  166  from `confTemplate/link.html` to `gitja/index.html` will be copied, keeping the link
  167  intact, resulting in a symbolic link at `output/link.html` essentially pointing to
  168  `output/gitja/index.html`.
  169  -}
  170  copyStaticDirs :: [Path Abs Dir] -> Path Abs Dir -> IO ()
  171  copyStaticDirs dirs output = mapM_ copy dirs
  172    where
  173      copy :: Path Abs Dir -> IO ()
  174      copy p = do
  175          let isRepo = (toFilePath . dirname $ p) == "repo/"
  176          isLink <- pathIsSymbolicLink . toFilePath $ p
  177          when (not isRepo || isLink) $ do
  178              let output' = output </> dirname p
  179              let fp = FP.dropTrailingPathSeparator . toFilePath $ p
  180              if isLink
  181                  then do
  182                      target <- getSymbolicLinkTarget fp
  183                      createDirectoryLink target . FP.dropTrailingPathSeparator . toFilePath $ output'
  184                  else copyDirRecur p output'
  185  
  186  copyStaticFiles :: [Path Abs File] -> Path Abs Dir -> IO ()
  187  copyStaticFiles files output = mapM_ copy files
  188    where
  189      copy :: Path Abs File -> IO ()
  190      copy p = do
  191          let fp = toFilePath p
  192          let isTemplate = FP.takeExtension fp `elem` [".html", ".include"]
  193          isLink <- pathIsSymbolicLink fp
  194          when (not isTemplate || isLink) $ do
  195              let output' = output </> filename p
  196              if isLink
  197                  then do
  198                      target <- getSymbolicLinkTarget fp
  199                      maybeExists <- forgivingAbsence . isSymlink $ output'
  200                      let exists = fromMaybe False maybeExists
  201                      when exists . removeFile . toFilePath $ output'
  202                      createFileLink target . toFilePath $ output'
  203                  else copyFile p output'
