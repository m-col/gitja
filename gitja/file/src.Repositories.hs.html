   49  {-
   50  This is the entrypoint that maps over our repositories, reading from them and writing
   51  out their web pages using the loaded templates.
   52  -}
   53  run :: Env -> IO [Repo]
   54  run env = do
   55      repos <- loadRepos env
   56      mapM (processRepo env repos) repos
   57  
   58  {-
   59  Get paths along with their descriptions.
   60  -}
   61  loadRepos :: Env -> IO [Repo]
   62  loadRepos = mapM mkRepo <=< filterM (fmap isRight . okRepo) . envRepos
   63    where
   64      okRepo :: Path Abs Dir -> IO (Either Git.GitException LgRepo)
   65      okRepo p =
   66          try . liftIO . Git.openRepository lgFactory $
   67              Git.defaultRepositoryOptions{Git.repoPath = toFilePath p}
   68  
   69      mkRepo :: Path Abs Dir -> IO Repo
   70      mkRepo p = (\d -> Repo p d Nothing) <$> getDescription p
   71  
   72  {-
   73  Pass the repository's folder, get its description. The approach is:
   74  
   75      1. Check for a description in repo/description
   76      2. Check for a description in repo/.git/description
   77      3. Fallback to the repo folder's name.
   78  
   79  -}
   80  getDescription :: Path Abs Dir -> IO T.Text
   81  getDescription dir =
   82      ifM
   83          (D.doesFileExist inTop)
   84          (return . Just $ inTop)
   85          ( ifM
   86              (D.doesFileExist inGit)
   87              (return . Just $ inGit)
   88              (return Nothing)
   89          )
   90          >>= \case
   91              Just file ->
   92                  T.strip . T.pack <$> readFile file
   93              Nothing ->
   94                  return . T.pack . toFilePath . dirname $ dir
   95    where
   96      inTop = toFilePath dir FP.</> "description"
   97      inGit = toFilePath dir FP.</> ".git" FP.</> "description"

   99  {-
  100  This receives a file path to a single repository and tries to process it. If the
  101  repository doesn't exist or is unreadable in any way we can forget about it and move on
  102  (after informing the user of course).
  103  -}
  104  processRepo :: Env -> [Repo] -> Repo -> IO Repo
  105  processRepo env repos repo =
  106      Git.withRepository lgFactory (toFilePath . repositoryPath $ repo) $ processRepo' env repos repo
  107  
  108  processRepo' :: Env -> [Repo] -> Repo -> ReaderT LgRepo IO Repo
  109  processRepo' env repos repo = do
  110      let name = dirname . repositoryPath $ repo
  111      let directory = envOutput env </> name
  112  
  113      Git.resolveReference "HEAD" >>= \case
  114          Nothing -> do
  115              liftIO . unless (envQuiet env) . putStrLn $ "gitja: " <> show name <> ": Failed to resolve HEAD."
  116              return repo
  117          Just commitID -> do
  118              let gitHead = Tagged commitID
  119              headCommit <- loadDiff =<< Git.lookupCommit gitHead
  120  
  121              -- If a page exists for the head commit, don't do anything else --
  122              exists <-
  123                  liftIO . D.doesFileExist $
  124                      toFilePath directory FP.</> "commit" FP.</> show commitID <> ".html"
  125  
  126              when (not exists || envForce env) $ do
  127                  -- Collect variables available to the ginger templates --
  128                  commits <- getCommits gitHead
  129                  tree <- getTree gitHead
  130                  let scope = package env repos name (repositoryDescription repo) commits tree
  131  
  132                  withRunInIO \runInIO -> do
  133                      -- Create the destination folders --
  134                      commitDir <- (directory </>) <$> parseRelDir "commit"
  135                      fileDir <- (directory </>) <$> parseRelDir "file"
  136                      ensureDir commitDir
  137                      ensureDir fileDir
  138  
  139                      -- Check which commits are new since the last run --
  140                      newCommits <- getUpdates commitDir commits
  141  
  142                      -- Run the generator --
  143                      let quiet = envQuiet env
  144                          force = envForce env
  145                          gen = genTarget scope runInIO quiet
  146  
  147                      mapM_ (generate scope runInIO quiet directory) (envRepoTemplates env)
  148  
  149                      whenJust (envCommitTemplate env) \commitT -> do
  150                          mapM_ (gen force commitT "commit" commitDir commitHref) newCommits
  151  
  152                      whenJust (envFileTemplate env) \fileT -> do
  153                          if force
  154                              then mapM_ (gen True fileT "file" fileDir fileHref) tree
  155                              else
  156                                  let updatedFiles = getUpdatedFiles tree newCommits
  157                                   in mapM_ (gen True fileT "file" fileDir fileHref) updatedFiles
  158  
  159                      -- Copy any static files/folders into the output directory --
  160                      envRepoCopyStatics env directory
  161  
  162              return
  163                  repo{repositoryHead = Just headCommit}
  164  
  165  {-
  166  The role of the function above is to gather information about a git repository and
  167  package it all together in such a way that various parts can be accessed and used by
  168  Ginger templates. `package` takes any pre-loaded information and places it into a
  169  hashmap for Ginger. Other non pre-loaded variables are looked up on the upon request.
  170  -}
  171  package ::
  172      Env ->
  173      [Repo] ->
  174      Path Rel Dir ->
  175      T.Text ->
  176      [Commit] ->
  177      [TreeFile] ->
  178      HashMap.HashMap T.Text (GVal RunRepo)
  179  package env repos name description commits tree =
  180      HashMap.fromList
  181          [ ("host", toGVal . envHost $ env)
  182          , ("repositories", toGVal repos)
  183          , ("name", toGVal . T.pack . init . toFilePath $ name)
  184          , ("description", toGVal description)
  185          , ("commits", toGVal commits)
  186          , ("tree", toGVal . filter (notElem FP.pathSeparator . T.unpack . treeFilePath) $ tree)
  187          , ("tree_recursive", toGVal tree)
  188          , ("readme", toGVal . findFile "readme" $ tree)
  189          , ("license", toGVal . findFile "license" $ tree)
  190          ]
  191    where
  192      -- Find a file in the tree starting with the specified prefix. The prefix is looked
  193      -- for on the full path, so will only find files in the top level directory.
  194      findFile :: T.Text -> [TreeFile] -> Maybe TreeFile
  195      findFile prefix = find (T.isPrefixOf prefix . T.toLower . treeFilePath)

  197  {-
  198  Collect commit history up to a head.
  199  -}
  200  getCommits :: Git.CommitOid LgRepo -> ReaderT LgRepo IO [Commit]
  201  getCommits commitID =
  202      fmap reverse . sequence . mapMaybe loadCommit
  203          <=< runConduit
  204          $ Git.sourceObjects Nothing commitID False .| sinkList
  205    where
  206      loadCommit :: Git.ObjectOid LgRepo -> Maybe (ReaderT LgRepo IO Commit)
  207      loadCommit (Git.CommitObjOid oid) = Just $ loadDiff =<< Git.lookupCommit oid
  208      loadCommit _ = Nothing
  209  
  210  {-
  211  Collect diff information for a single commit.
  212  -}
  213  loadDiff :: Git.Commit LgRepo -> ReaderT LgRepo IO Commit
  214  loadDiff gitCommit = do
  215      newTree <- Git.lookupTree . Git.commitTree $ gitCommit
  216      oldTree <-
  217          case listToMaybe . Git.commitParents $ gitCommit of
  218              Just parent ->
  219                  fmap Just . Git.lookupTree . Git.commitTree =<< Git.lookupCommit parent
  220              Nothing ->
  221                  return Nothing
  222  
  223      ioref <- liftIO . newIORef $ []
  224      lgDiffTreeToTree
  225          (fileCallback ioref)
  226          (hunkCallback ioref)
  227          (dataCallback ioref)
  228          oldTree
  229          (Just newTree)
  230      diffs <- liftIO . readIORef $ ioref
  231  
  232      return . Commit gitCommit . fmap fixupLists $ diffs
  233    where
  234      fileCallback ::
  235          IORef [Diff] ->
  236          Ptr LG.C'git_diff_delta ->
  237          CFloat ->
  238          Ptr () ->
  239          IO CInt
  240      fileCallback ioref dPtr _progress _payload = do
  241          delta <- peek dPtr
  242          newFile <- B.packCString . LG.c'git_diff_file'path . LG.c'git_diff_delta'new_file $ delta
  243          oldFile <- B.packCString . LG.c'git_diff_file'path . LG.c'git_diff_delta'old_file $ delta
  244          let oldFile' = bool Nothing (Just oldFile) (newFile /= oldFile)
  245              status = toEnum . fromIntegral . LG.c'git_diff_delta'status $ delta
  246              diff = Diff newFile oldFile' status []
  247          modifyIORef ioref (diff :)
  248          return 0
  249  
  250      hunkCallback ::
  251          IORef [Diff] ->
  252          Ptr LG.C'git_diff_delta ->
  253          Ptr LG.C'git_diff_range ->
  254          CString ->
  255          CSize ->
  256          Ptr () ->
  257          IO CInt
  258      hunkCallback ioref _dPtr _range header headerLen _payload = do
  259          (cur : _, rest) <- splitAt 1 <$> readIORef ioref
  260          bs <- curry B.packCStringLen header (fromIntegral headerLen)
  261          let hunk = Hunk bs []
  262          writeIORef ioref $ cur{diffHunks = hunk : diffHunks cur} : rest
  263          return 0
  264  
  265      dataCallback ::
  266          IORef [Diff] ->
  267          Ptr LG.C'git_diff_delta ->
  268          Ptr LG.C'git_diff_range ->
  269          CChar ->
  270          CString ->
  271          CSize ->
  272          Ptr () ->
  273          IO CInt
  274      dataCallback ioref _dPtr _range lineOrigin content contentLen _payload = do
  275          bs <- curry B.packCStringLen content (fromIntegral contentLen)
  276          let bs' = B.cons (fromIntegral lineOrigin) bs
  277          (cur : _, rest) <- splitAt 1 <$> readIORef ioref
  278          let (curHunk : _, restHunks) = splitAt 1 . diffHunks $ cur
  279          let updated =
  280                  cur
  281                      { diffHunks =
  282                          curHunk
  283                              { hunkLines = bs' : hunkLines curHunk
  284                              } :
  285                          restHunks
  286                      }
  287          writeIORef ioref $ updated : rest
  288          return 0
  289  
  290      -- The callbacks prepend and then we put the lists the right way round here.
  291      -- This avoids traversing the lists every time we add an item.
  292      fixupLists :: Diff -> Diff
  293      fixupLists diff = diff{diffHunks = fmap fixupLines . reverse . diffHunks $ diff}
  294        where
  295          fixupLines hunk = hunk{hunkLines = reverse . hunkLines $ hunk}

  297  {-
  298  Collect tree information for the given commit. Recurses on directories to list their
  299  contents.
  300  -}
  301  getTree :: Git.CommitOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  302  getTree = getTree' "" 0 . Git.commitTree <=< Git.lookupCommit
  303    where
  304      getTree' :: Git.TreeFilePath -> Int -> Git.TreeOid LgRepo -> ReaderT LgRepo IO [TreeFile]
  305      getTree' parent count toid = do
  306          one <- Git.lookupTree toid
  307          entries <- Git.listTreeEntries one
  308          let entries' = fmap (prependParent parent) entries
  309          contents <- mapM (\x -> getEntryContents x (count + 1)) entries'
  310          modes <- mapM (getEntryModes . snd) entries'
  311          return $ zipWith3 TreeFile (fmap treePaths entries') contents modes
  312  
  313      prependParent ::
  314          Git.TreeFilePath ->
  315          (Git.TreeFilePath, Git.TreeEntry r) ->
  316          (Git.TreeFilePath, Git.TreeEntry r)
  317      prependParent "" pathentry = pathentry
  318      prependParent parent (path, entry) = (mconcat [parent, "/", path], entry)
  319  
  320      getEntryContents :: (Git.TreeFilePath, Git.TreeEntry LgRepo) -> Int -> ReaderT LgRepo IO TreeFileContents
  321      getEntryContents (_, Git.BlobEntry oid _) _ = getBlobContents oid
  322      getEntryContents (path, Git.TreeEntry oid) count = FolderContents <$> getTree' path count oid
  323      getEntryContents (_, Git.CommitEntry oid) _ = return . FileContents . B.fromString . show . untag $ oid
  324  
  325      getEntryModes :: Git.TreeEntry r -> ReaderT r IO TreeEntryMode
  326      getEntryModes (Git.BlobEntry _ kind) = return . blobkindToMode $ kind
  327      getEntryModes (Git.TreeEntry _) = return ModeDirectory
  328      getEntryModes (Git.CommitEntry _) = return ModeSubmodule
  329  
  330      treePaths :: (Git.TreeFilePath, Git.TreeEntry r) -> T.Text
  331      treePaths = T.decodeUtf8With T.lenientDecode . fst

  333  {-
  334  Collect information about references. TODO: Find a more canonical way to split
  335  references into tags or branches rather than filtering the refnames.
  336  -}
  337  getRefs :: T.Text -> ReaderT LgRepo IO [Ref]
  338  getRefs ref = do
  339      names <- filter (T.isPrefixOf ref) <$> Git.listReferences
  340      maybeOids <- mapM Git.resolveReference names
  341      let names' = catMaybes . zipWith dropName maybeOids $ names
  342      objs <- mapM Git.lookupObject . catMaybes $ maybeOids
  343      maybeCommits <- mapM refObjToCommit objs
  344      let names'' = map (fromJust . T.stripPrefix ref) . catMaybes . zipWith dropName maybeCommits $ names'
  345      return . zipWith Ref names'' . catMaybes $ maybeCommits
  346    where
  347      refObjToCommit ::
  348          Git.Object LgRepo (ReaderT LgRepo IO) ->
  349          ReaderT LgRepo IO (Maybe Commit)
  350      refObjToCommit (Git.CommitObj obj) = Just <$> loadDiff obj
  351      refObjToCommit _ = return Nothing
  352  
  353      dropName :: Maybe a -> Git.RefName -> Maybe Git.RefName
  354      dropName (Just _) name = Just name
  355      dropName Nothing _ = Nothing

  357  {-
  358  Get the list of commits that need updating since the last run. New commits are
  359  identified by the absence of their output file.
  360  -}
  361  getUpdates :: Path Abs Dir -> [Commit] -> IO [Commit]
  362  getUpdates _ [] = return []
  363  getUpdates directory cs = go cs
  364    where
  365      dir = toFilePath directory
  366  
  367      go :: [Commit] -> IO [Commit]
  368      go [] = return []
  369      go (x : xs) =
  370          ifM
  371              (D.doesFileExist $ dir FP.</> commitHref x)
  372              (return [])
  373              ((x :) <$> go xs)
  374  
  375  getUpdatedFiles :: [TreeFile] -> [Commit] -> [TreeFile]
  376  getUpdatedFiles [] _ = []
  377  getUpdatedFiles _ [] = []
  378  getUpdatedFiles files commits = filter ((`elem` updated) . treeFilePath) files
  379    where
  380      updated :: [T.Text]
  381      updated = fmap (bsToText . diffNewFile) . concatMap commitDiffs $ commits

  383  {-
  384  Render repository data into a template and save it to file.
  385  -}
  386  generate ::
  387      HashMap.HashMap T.Text (GVal RunRepo) ->
  388      (ReaderT LgRepo IO (GVal RunRepo) -> IO (GVal RunRepo)) ->
  389      Bool ->
  390      Path Abs Dir ->
  391      Template ->
  392      IO ()
  393  generate scope runInIO quiet directory template = do
  394      let output = toFilePath (directory </> templatePath template)
  395      unless quiet . putStrLn $ "Writing " <> output
  396      TL.writeFile output =<< render (cbRepoLookup scope runInIO) template
  397  
  398  {-
  399  A Target refers to a template scope and repository object whose information is available
  400  in that scope. For example, commits are a target as they each generate a scope
  401  containing that commit's information, and these scopes are each rendered in the
  402  commitTemplate.
  403  -}
  404  genTarget ::
  405      ToGVal RunRepo t =>
  406      HashMap.HashMap T.Text (GVal RunRepo) ->
  407      (ReaderT LgRepo IO (GVal RunRepo) -> IO (GVal RunRepo)) ->
  408      Bool ->
  409      Bool ->
  410      Template ->
  411      T.Text ->
  412      Path Abs Dir ->
  413      (t -> FilePath) ->
  414      t ->
  415      IO ()
  416  genTarget scope runInIO quiet force template category directory href target = do
  417      output <- fmap (directory </>) . parseRelFile . href $ target
  418      exists <- doesFileExist output
  419      when (force || not exists) $ do
  420          let output' = toFilePath output
  421              scope' = HashMap.insert category (toGVal target) scope
  422          unless quiet . putStrLn $ "Writing " <> output'
  423          TL.writeFile output' =<< render (cbRepoLookup scope' runInIO) template
  424  
  425  commitHref :: Commit -> FilePath
  426  commitHref = (++ ".html") . commitHash
  427  
  428  fileHref :: TreeFile -> FilePath
  429  fileHref = T.unpack . treePathToHref

  431  {-
  432  With a dictionary of preloaded values and a function to access additional data, create a
  433  lookup function for rendering.
  434  -}
  435  cbRepoLookup ::
  436      HashMap.HashMap T.Text (GVal RunRepo) ->
  437      (ReaderT LgRepo IO (GVal RunRepo) -> IO (GVal RunRepo)) ->
  438      T.Text ->
  439      RunRepo (GVal RunRepo)
  440  cbRepoLookup scope runInIO key = liftIO . runInIO $ case key of
  441      "tags" -> toGVal <$> getRefs "refs/tags/"
  442      "branches" -> toGVal <$> getRefs "refs/heads/"
  443      key' -> return . toGVal . HashMap.lookup key' $ scope
